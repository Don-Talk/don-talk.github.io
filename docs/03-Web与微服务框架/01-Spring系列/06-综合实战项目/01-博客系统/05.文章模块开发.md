### 4.2 ArticleServiceImpl.java

```java
package com.dontalk.blog.module.article.service.impl;

import com.baomidou.mybatisplus.core.conditions.query.LambdaQueryWrapper;
import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.dontalk.blog.common.exception.BusinessException;
import com.dontalk.blog.module.article.dto.ArticleCreateRequest;
import com.dontalk.blog.module.article.dto.ArticleQueryRequest;
import com.dontalk.blog.module.article.dto.ArticleUpdateRequest;
import com.dontalk.blog.module.article.entity.Article;
import com.dontalk.blog.module.article.mapper.ArticleMapper;
import com.dontalk.blog.module.article.service.ArticleService;
import com.dontalk.blog.module.article.vo.ArticleResponse;
import com.dontalk.blog.module.category.entity.Category;
import com.dontalk.blog.module.category.mapper.CategoryMapper;
import com.dontalk.blog.module.tag.entity.Tag;
import com.dontalk.blog.module.tag.mapper.TagMapper;
import com.dontalk.blog.module.user.entity.User;
import com.dontalk.blog.module.user.mapper.UserMapper;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.BeanUtils;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import org.springframework.util.StringUtils;

import java.util.Arrays;
import java.util.List;
import java.util.stream.Collectors;

@Slf4j
@Service
@RequiredArgsConstructor
public class ArticleServiceImpl implements ArticleService {
    
    private final ArticleMapper articleMapper;
    private final CategoryMapper categoryMapper;
    private final TagMapper tagMapper;
    private final UserMapper userMapper;
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public Long createArticle(ArticleCreateRequest request, Long authorId) {
        // 1. 验证分类是否存在
        Category category = categoryMapper.selectById(request.getCategoryId());
        if (category == null) {
            throw new BusinessException("分类不存在");
        }
        
        // 2. 构建文章实体
        Article article = new Article();
        BeanUtils.copyProperties(request, article);
        article.setAuthorId(authorId);
        article.setViewCount(0);
        article.setLikeCount(0);
        article.setCommentCount(0);
        
        // 3. 保存文章
        articleMapper.insert(article);
        
        // 4. 保存文章标签关联
        if (request.getTagIds() != null && request.getTagIds().length > 0) {
            saveArticleTags(article.getId(), Arrays.asList(request.getTagIds()));
        }
        
        log.info("创建文章成功: id={}, title={}", article.getId(), article.getTitle());
        return article.getId();
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void updateArticle(ArticleUpdateRequest request) {
        // 1. 查询文章是否存在
        Article article = articleMapper.selectById(request.getId());
        if (article == null) {
            throw new BusinessException("文章不存在");
        }
        
        // 2. 验证分类是否存在
        Category category = categoryMapper.selectById(request.getCategoryId());
        if (category == null) {
            throw new BusinessException("分类不存在");
        }
        
        // 3. 更新文章
        BeanUtils.copyProperties(request, article);
        articleMapper.updateById(article);
        
        // 4. 更新标签关联(先删除再新增)
        if (request.getTagIds() != null) {
            deleteArticleTags(article.getId());
            if (request.getTagIds().length > 0) {
                saveArticleTags(article.getId(), Arrays.asList(request.getTagIds()));
            }
        }
        
        log.info("更新文章成功: id={}", article.getId());
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void deleteArticle(Long id) {
        Article article = articleMapper.selectById(id);
        if (article == null) {
            throw new BusinessException("文章不存在");
        }
        
        // 逻辑删除: 设置状态为已删除
        article.setStatus(2);
        articleMapper.updateById(article);
        
        log.info("删除文章成功: id={}", id);
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void publishArticle(Long id) {
        Article article = articleMapper.selectById(id);
        if (article == null) {
            throw new BusinessException("文章不存在");
        }
        
        // 发布文章: 设置状态为已发布
        article.setStatus(1);
        articleMapper.updateById(article);
        
        log.info("发布文章成功: id={}", id);
    }
    
    @Override
    public ArticleResponse getArticleById(Long id) {
        // 1. 查询文章
        Article article = articleMapper.selectById(id);
        if (article == null || article.getStatus() == 2) {
            throw new BusinessException("文章不存在或已删除");
        }
        
        // 2. 转换为VO
        ArticleResponse response = convertToResponse(article);
        
        // 3. 填充作者信息
        User author = userMapper.selectById(article.getAuthorId());
        if (author != null) {
            response.setAuthorName(author.getNickname());
        }
        
        // 4. 填充分类信息
        Category category = categoryMapper.selectById(article.getCategoryId());
        if (category != null) {
            response.setCategoryName(category.getName());
        }
        
        // 5. 填充标签信息
        List<ArticleResponse.TagInfo> tags = getArticleTags(article.getId());
        response.setTags(tags);
        
        return response;
    }
    
    @Override
    public Page<ArticleResponse> queryArticles(ArticleQueryRequest request) {
        // 1. 构建分页对象
        Page<Article> page = new Page<>(request.getPageNum(), request.getPageSize());
        
        // 2. 构建查询条件
        LambdaQueryWrapper<Article> wrapper = new LambdaQueryWrapper<>();
        
        // 关键词搜索
        if (StringUtils.hasText(request.getKeyword())) {
            wrapper.and(w -> w.like(Article::getTitle, request.getKeyword())
                    .or()
                    .like(Article::getSummary, request.getKeyword())
                    .or()
                    .like(Article::getContent, request.getKeyword()));
        }
        
        // 分类筛选
        if (request.getCategoryId() != null) {
            wrapper.eq(Article::getCategoryId, request.getCategoryId());
        }
        
        // 作者筛选
        if (request.getAuthorId() != null) {
            wrapper.eq(Article::getAuthorId, request.getAuthorId());
        }
        
        // 状态筛选
        if (request.getStatus() != null) {
            wrapper.eq(Article::getStatus, request.getStatus());
        } else {
            // 默认只查询已发布的文章
            wrapper.eq(Article::getStatus, 1);
        }
        
        // 排除已删除的文章
        wrapper.ne(Article::getStatus, 2);
        
        // 排序
        String orderBy = request.getOrderBy();
        if ("view_count".equals(orderBy)) {
            wrapper.orderByDesc(Article::getViewCount);
        } else if ("like_count".equals(orderBy)) {
            wrapper.orderByDesc(Article::getLikeCount);
        } else {
            wrapper.orderByDesc(Article::getCreateTime);
        }
        
        // 3. 执行查询
        Page<Article> articlePage = articleMapper.selectPage(page, wrapper);
        
        // 4. 转换为VO
        Page<ArticleResponse> responsePage = new Page<>();
        BeanUtils.copyProperties(articlePage, responsePage);
        responsePage.setRecords(articlePage.getRecords().stream()
                .map(this::convertToSimpleResponse)
                .collect(Collectors.toList()));
        
        return responsePage;
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void incrementViewCount(Long id) {
        articleMapper.incrementViewCount(id);
    }
    
    @Override
    @Transactional(rollbackFor = Exception.class)
    public void likeArticle(Long id) {
        articleMapper.incrementLikeCount(id);
    }
    
    @Override
    public Page<ArticleResponse> searchArticles(String keyword, Integer pageNum, Integer pageSize) {
        ArticleQueryRequest request = new ArticleQueryRequest();
        request.setKeyword(keyword);
        request.setPageNum(pageNum);
        request.setPageSize(pageSize);
        request.setStatus(1); // 只搜索已发布的文章
        return queryArticles(request);
    }
    
    // ==================== 私有方法 ====================
    
    /**
     * 转换为完整响应VO
     */
    private ArticleResponse convertToResponse(Article article) {
        ArticleResponse response = new ArticleResponse();
        BeanUtils.copyProperties(article, response);
        return response;
    }
    
    /**
     * 转换为简单响应VO(不含内容)
     */
    private ArticleResponse convertToSimpleResponse(Article article) {
        ArticleResponse response = convertToResponse(article);
        // 列表页不返回完整内容,减少数据传输
        response.setContent(null);
        return response;
    }
```

```java
package com.dontalk.blog.module.article.controller;

import com.baomidou.mybatisplus.extension.plugins.pagination.Page;
import com.dontalk.blog.common.result.Result;
import com.dontalk.blog.module.article.dto.ArticleCreateRequest;
import com.dontalk.blog.module.article.dto.ArticleQueryRequest;
import com.dontalk.blog.module.article.dto.ArticleUpdateRequest;
import com.dontalk.blog.module.article.service.ArticleService;
import com.dontalk.blog.module.article.vo.ArticleResponse;
import com.dontalk.blog.security.util.SecurityUtils;
import io.swagger.annotations.Api;
import io.swagger.annotations.ApiOperation;
import lombok.RequiredArgsConstructor;
import org.springframework.validation.annotation.Validated;
import org.springframework.web.bind.annotation.*;

@RestController
@RequestMapping("/api/articles")
@RequiredArgsConstructor
@Api(tags = "文章管理")
public class ArticleController {
    
    private final ArticleService articleService;
    
    @PostMapping
    @ApiOperation("创建文章")
    public Result<Long> createArticle(@Validated @RequestBody ArticleCreateRequest request) {
        Long authorId = SecurityUtils.getCurrentUserId();

        return Result.success(articleId);
    }
    
    @PutMapping
    @ApiOperation("更新文章")

        articleService.updateArticle(request);
        return Result.success();

        return Result.success();
    }
    
    @PostMapping("/{id}/publish")
    @ApiOperation("发布文章")
    public Result<Void> publishArticle(@PathVariable Long id) {
        articleService.publishArticle(id);
        return Result.success();
    }
    
    @GetMapping("/{id}")
    @ApiOperation("根据ID查询文章")
    public Result<ArticleResponse> getArticle(@PathVariable Long id) {
```

## 八、常见问题

### Q1: 如何实现文章的草稿箱功能?

**A**: 通过`status`字段区分:
- `status=0`: 草稿
- `status=1`: 已发布
- `status=2`: 已删除

用户点击"保存草稿"时设置status=0，点击"发布"时设置status=1。

### Q2: 如何优化大量文章的查询性能?

**A**: 
1. **索引优化**: 在`category_id`、`status`、`create_time`字段建立联合索引
2. **缓存**: 使用Redis缓存热门文章
3. **分页**: 避免一次性加载所有数据
4. **读写分离**: 主库写、从库读

### Q3: 如何实现文章的定时发布?

**A**: 使用Spring的`@Scheduled`定时任务:
```java
@Scheduled(fixedRate = 60000) // 每分钟执行一次
public void publishScheduledArticles() {
    // 查询到达发布时间但未发布的文章
    List<Article> articles = articleMapper.selectList(
        new LambdaQueryWrapper<Article>()
            .eq(Article::getStatus, 0)
            .le(Article::getPublishTime, LocalDateTime.now())
    );
    
    // 批量发布
    articles.forEach(article -> {
        article.setStatus(1);
        articleMapper.updateById(article);
    });
}
```
