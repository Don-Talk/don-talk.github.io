## 1. 拦截器（Interceptor）

### 1.1 什么是拦截器？

拦截器允许在 Hibernate 操作的各个阶段插入自定义逻辑，如保存前、更新后等。

### 1.2 实现拦截器

```java
package com.example.interceptor;

import org.hibernate.EmptyInterceptor;
import org.hibernate.type.Type;

import java.io.Serializable;
import java.util.Date;

public class AuditInterceptor extends EmptyInterceptor {
    
    /**
     * 保存前调用
     */
    @Override
    public boolean onSave(Object entity, Serializable id, Object[] state,
                         String[] propertyNames, Type[] types) {
        
        if (entity instanceof Auditable) {
            Auditable auditable = (Auditable) entity;
            auditable.setCreateTime(new Date());
            auditable.setUpdateTime(new Date());
            
            // 找到对应的属性索引并修改
            for (int i = 0; i < propertyNames.length; i++) {
                if ("createTime".equals(propertyNames[i])) {
                    state[i] = new Date();
                }
                if ("updateTime".equals(propertyNames[i])) {
                    state[i] = new Date();
                }
            }
            return true;  // 返回 true 表示状态已修改
        }
        
        return false;
    }
    
    /**
     * 更新前调用
     */
    @Override
    public boolean onFlushDirty(Object entity, Serializable id, Object[] currentState,
                               Object[] previousState, String[] propertyNames, Type[] types) {
        
        if (entity instanceof Auditable) {
            for (int i = 0; i < propertyNames.length; i++) {
                if ("updateTime".equals(propertyNames[i])) {
                    currentState[i] = new Date();
                }
            }
            return true;
        }
        
        return false;
    }
}
```

### 1.3 注册拦截器

#### 方式一：SessionFactory 级别

```xml
<!-- hibernate.cfg.xml -->
<property name="hibernate.session_factory.interceptor">
    com.example.interceptor.AuditInterceptor
</property>
```

#### 方式二：Session 级别

```java
Session session = sessionFactory.withOptions()
    .interceptor(new AuditInterceptor())
    .openSession();
```

## 2. 事件监听器（EventListener）

### 2.1 事件类型

| 事件 | 说明 |
|------|------|
| PreInsertEvent | 插入前 |
| PostInsertEvent | 插入后 |
| PreUpdateEvent | 更新前 |
| PostUpdateEvent | 更新后 |
| PreDeleteEvent | 删除前 |
| PostDeleteEvent | 删除后 |
| LoadEvent | 加载时 |

### 2.2 实现监听器

```java
package com.example.listener;

import org.hibernate.event.spi.PreInsertEvent;
import org.hibernate.event.spi.PreInsertEventListener;

import java.util.Date;

public class AuditListener implements PreInsertEventListener {
    
    @Override
    public boolean onPreInsert(PreInsertEvent event) {
        Object entity = event.getEntity();
        
        if (entity instanceof Auditable) {
            Date now = new Date();
            ((Auditable) entity).setCreateTime(now);
            ((Auditable) entity).setUpdateTime(now);
            return false;
        }
        
        return false;
    }
}
```

### 2.3 注册监听器

```xml
<property name="hibernate.event.pre_insert">
    com.example.listener.AuditListener
</property>
```

## 3. 自定义类型（UserType）

### 3.1 场景：加密字段

```java
package com.example.type;

import org.hibernate.engine.spi.SharedSessionContractImplementor;
import org.hibernate.usertype.UserType;

import java.io.Serializable;
import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.sql.SQLException;
import java.sql.Types;

public class EncryptedStringType implements UserType {
    
    @Override
    public int[] sqlTypes() {
        return new int[]{Types.VARCHAR};
    }
    
    @Override
    public Class<?> returnedClass() {
        return String.class;
    }
    
    @Override
    public Object nullSafeGet(ResultSet rs, String[] names,
                             SharedSessionContractImplementor session, Object owner)
                             throws SQLException {
        String value = rs.getString(names[0]);
        if (value != null) {
            return decrypt(value);  // 解密
        }
        return null;
    }
    
    @Override
    public void nullSafeSet(PreparedStatement st, Object value, int index,
                           SharedSessionContractImplementor session)
                           throws SQLException {
        if (value != null) {
            st.setString(index, encrypt((String) value));  // 加密
        } else {
            st.setNull(index, Types.VARCHAR);
        }
    }
    
    private String encrypt(String plainText) {
        // 加密逻辑
        return "encrypted_" + plainText;
    }
    
    private String decrypt(String encryptedText) {
        // 解密逻辑
        return encryptedText.replace("encrypted_", "");
    }
    
    // 其他方法省略...
}
```

### 3.2 使用自定义类型

```java
@Entity
public class User {
    
    @Id
    @GeneratedValue
    private Long id;
    
    @Type(type = "com.example.type.EncryptedStringType")
    private String password;  // 自动加密存储
}
```

## 4. 原生 SQL 查询

### 4.1 基本用法

```java
@Test
public void testNativeSQL() {
    Session session = sessionFactory.openSession();
    
    // 简单查询
    List<Object[]> results = session.createNativeQuery(
            "SELECT id, username, email FROM user WHERE age > :age")
            .setParameter("age", 20)
            .list();
    
    session.close();
}
```

### 4.2 映射到实体

```java

public void testNativeSQLEntity() {

            .setParameter("age", 20)
            .list();
    
    session.close();
}
```

## 5. 存储过程调用

### 5.1 创建存储过程

```sql
DELIMITER //
CREATE PROCEDURE GetUserCount(IN minAge INT, OUT count INT)
BEGIN
    SELECT COUNT(*) INTO count FROM user WHERE age > minAge;
END //
DELIMITER ;
```

### 5.2 调用存储过程

```java
@Test
public void testStoredProcedure() {
    Session session = sessionFactory.openSession();
    Transaction tx = session.beginTransaction();
    
    // 方式一：使用 createStoredProcedureQuery
    StoredProcedureQuery query = session.createStoredProcedureQuery("GetUserCount");
    query.registerStoredProcedureParameter("minAge", Integer.class, ParameterMode.IN);
    query.registerStoredProcedureParameter("count", Integer.class, ParameterMode.OUT);
    
    query.setParameter("minAge", 20);
    query.execute();
    
    Integer count = (Integer) query.getOutputParameterValue("count");
    System.out.println("用户数量：" + count);
    
    tx.commit();
    session.close();
}
```

## 6. 批量操作优化

### 6.1 批量插入

```java
@Test
public void testBatchInsert() {
    Session session = sessionFactory.openSession();
    Transaction tx = session.beginTransaction();
    
    int batchSize = 50;
    for (int i = 0; i < 1000; i++) {
        User user = new User();
        user.setUsername("用户" + i);
        session.save(user);
        
        if (i % batchSize == 0 && i > 0) {
            session.flush();
            session.clear();
        }
    }
    
    tx.commit();
    session.close();
}
```

**配置批量大小**：

```xml
<property name="hibernate.jdbc.batch_size">50</property>
<property name="hibernate.order_inserts">true</property>
<property name="hibernate.order_updates">true</property>
```

### 6.2 批量更新

```java
@Test
public void testBatchUpdate() {
    Session session = sessionFactory.openSession();
    Transaction tx = session.beginTransaction();
    
    // HQL 批量更新
    int affectedRows = session.createQuery(
            "UPDATE User SET status = :status WHERE age < :age")
            .setParameter("status", 0)
            .setParameter("age", 18)
            .executeUpdate();
    
    System.out.println("更新了 " + affectedRows + " 条记录");
    
    tx.commit();
    session.close();
}
```

## 7. 审计功能（Envers）

### 7.1 添加依赖

```xml
<dependency>
    <groupId>org.hibernate.orm</groupId>
    <artifactId>hibernate-envers</artifactId>
    <version>6.2.7.Final</version>
</dependency>
```

### 7.2 启用审计

```java
@Entity
@Audited  // 启用审计
public class User {
    
    @Id
    @GeneratedValue
    private Long id;
    
    private String username;
    
    private String email;
}
```

### 7.3 自动创建审计表

Hibernate Envers 会自动创建 `user_aud` 表，记录所有变更历史。

### 7.4 查询审计数据

```java
@Test
public void testAudit() {
    Session session = sessionFactory.openSession();
    
    AuditReader auditReader = AuditReaderFactory.get(session);
    
    // 查询某个版本的用户
    User user = auditReader.find(User.class, 1L, 1);  // 第 1 个版本
    
    // 查询所有修订版本
    List<Number> revisions = auditReader.getRevisions(User.class, 1L);
    
    // 查询某个时间的版本
    Date date = new Date();
    User userAtDate = auditReader.find(User.class, 1L, date);
    
    session.close();
}
```

## 8. 过滤器（Filter）

### 8.1 定义过滤器

```java
@Entity
@FilterDef(name = "statusFilter", parameters = @ParamDef(name = "status", type = "int"))
@Filters({
    @Filter(name = "statusFilter", condition = "status = :status")
})
public class User {
    
    @Id
    @GeneratedValue
    private Long id;
    
    private String username;
    
    private Integer status;  // 0-禁用，1-启用
}
```

### 8.2 启用过滤器

```java
@Test
public void testFilter() {
    Session session = sessionFactory.openSession();
    
    // 启用过滤器
    session.enableFilter("statusFilter").setParameter("status", 1);
    
    // 只查询启用的用户
    List<User> users = session.createQuery("FROM User", User.class).list();
    
    session.close();
}
```

## 9. 完整示例

### 9.1 审计拦截器 + 批量操作

```java
@Service
public class UserService {
    
    @Autowired
    private SessionFactory sessionFactory;
    
    @Transactional
    public void batchCreateUsers(List<User> users) {
        Session session = sessionFactory.getCurrentSession();
        
        int batchSize = 50;
        for (int i = 0; i < users.size(); i++) {
            session.save(users.get(i));
            
            if (i % batchSize == 0 && i > 0) {
                session.flush();
                session.clear();
            }
        }
    }
    
    public List<User> findActiveUsers() {
        Session session = sessionFactory.getCurrentSession();
        
        // 启用过滤器
        session.enableFilter("statusFilter").setParameter("status", 1);
        
        return session.createQuery("FROM User", User.class).list();
    }
}
```
