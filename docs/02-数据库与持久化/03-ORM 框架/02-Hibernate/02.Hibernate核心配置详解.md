**学习目标：**
> * 掌握 hibernate.cfg.xml 的完整结构
> * 理解数据库方言、连接池等核心配置

> * 了解二级缓存和日志配置

## 1. hibernate.cfg.xml 完整结构

### 1.1 配置文件 DTD

```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-configuration PUBLIC
        "-//Hibernate/Hibernate Configuration DTD 3.0//EN"
        "http://www.hibernate.org/dtd/hibernate-configuration-3.0.dtd">
```

### 1.2 配置项分类

```xml
<hibernate-configuration>
    <session-factory>
        <!-- 1. 数据库连接配置 -->
        <!-- 2. Hibernate 基本配置 -->
        <!-- 3. 缓存配置 -->
        <!-- 4. 日志配置 -->
        <!-- 5. 实体类映射 -->
    </session-factory>
</hibernate-configuration>
```

## 2. 数据库连接配置

### 2.1 基本连接配置

```xml
<!-- JDBC 驱动 -->
<property name="hibernate.connection.driver_class">com.mysql.cj.jdbc.Driver</property>

<!-- 数据库 URL -->
<property name="hibernate.connection.url">
    jdbc:mysql://localhost:3306/hibernate_demo?useSSL=false&amp;serverTimezone=Asia/Shanghai
</property>

<!-- 用户名和密码 -->
<property name="hibernate.connection.username">root</property>
<property name="hibernate.connection.password">123456</property>
```

### 2.2 使用外部配置文件

创建 `jdbc.properties`：

```properties
jdbc.driver=com.mysql.cj.jdbc.Driver
jdbc.url=jdbc:mysql://localhost:3306/hibernate_demo
jdbc.username=root
jdbc.password=123456
```

在 `hibernate.cfg.xml` 中引用：

```xml
<hibernate-configuration>
    <session-factory>
        <!-- 引用外部属性文件 -->
        <property name="hibernate.connection.driver_class">${jdbc.driver}</property>
        <property name="hibernate.connection.url">${jdbc.url}</property>
        <property name="hibernate.connection.username">${jdbc.username}</property>
        <property name="hibernate.connection.password">${jdbc.password}</property>
    </session-factory>
</hibernate-configuration>
```

## 3. 数据库方言（Dialect）

### 3.1 什么是方言？

方言告诉 Hibernate 如何生成特定数据库的 SQL 语句。

### 3.2 常用数据库方言

| 数据库 | 方言类 |
|--------|--------|
| MySQL 5.x | `org.hibernate.dialect.MySQL5Dialect` |
| MySQL 8.x | `org.hibernate.dialect.MySQLDialect` |
| Oracle | `org.hibernate.dialect.OracleDialect` |
| PostgreSQL | `org.hibernate.dialect.PostgreSQLDialect` |
| SQL Server | `org.hibernate.dialect.SQLServerDialect` |
| H2 | `org.hibernate.dialect.H2Dialect` |

### 3.3 配置示例

```xml
<!-- MySQL 8 -->
<property name="hibernate.dialect">org.hibernate.dialect.MySQLDialect</property>

<!-- Oracle -->
<property name="hibernate.dialect">org.hibernate.dialect.OracleDialect</property>
```

## 4. 连接池配置

### 4.1 Hibernate 内置连接池（不推荐生产使用）

```xml
<property name="hibernate.connection.pool_size">10</property>
```

### 4.2 C3P0 连接池

**添加依赖**：

```xml
<dependency>
    <groupId>org.hibernate.orm</groupId>
    <artifactId>hibernate-c3p0</artifactId>
    <version>6.2.7.Final</version>
</dependency>
```

**配置**：

```xml
<!-- 指定 C3P0 提供者 -->
<property name="hibernate.connection.provider_class">
    org.hibernate.c3p0.internal.C3P0ConnectionProvider
</property>

<!-- 最小连接数 -->
<property name="hibernate.c3p0.min_size">5</property>

<!-- 最大连接数 -->
<property name="hibernate.c3p0.max_size">20</property>

<!-- 超时时间（秒） -->
<property name="hibernate.c3p0.timeout">300</property>

<!-- 最大空闲时间（秒） -->
<property name="hibernate.c3p0.max_idle_time">3000</property>

<!-- 测试查询 -->
<property name="hibernate.c3p0.preferred_test_query">SELECT 1</property>
```

### 4.3 HikariCP 连接池（推荐）

**添加依赖**：

```xml
<dependency>
    <groupId>com.zaxxer</groupId>
    <artifactId>HikariCP</artifactId>
    <version>5.0.1</version>
</dependency>
```

**配置**：

```xml
<property name="hibernate.connection.provider_class">
    com.zaxxer.hikari.hibernate.HikariConnectionProvider
</property>

<property name="hibernate.hikari.minimumIdle">5</property>
<property name="hibernate.hikari.maximumPoolSize">20</property>
<property name="hibernate.hikari.idleTimeout">300000</property>
<property name="hibernate.hikari.connectionTimeout">20000</property>
<property name="hibernate.hikari.maxLifetime">1200000</property>
```

## 5. 实体类映射配置

### 5.1 注解方式（推荐）

```xml
<!-- 注册单个实体类 -->
<mapping class="com.example.entity.User"/>
<mapping class="com.example.entity.Product"/>

<!-- 或者扫描包（Hibernate 6+） -->
<mapping package="com.example.entity"/>
```

### 5.2 XML 方式

创建 `User.hbm.xml`：

```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-mapping PUBLIC
        "-//Hibernate/Hibernate Mapping DTD 3.0//EN"
        "http://www.hibernate.org/dtd/hibernate-mapping-3.0.dtd">

<hibernate-mapping>
    <class name="com.example.entity.User" table="user">
        <id name="id" column="id">
            <generator class="identity"/>
        </id>
        <property name="username" column="username"/>
        <property name="email" column="email"/>
        <property name="age" column="age"/>
    </class>
</hibernate-mapping>
```

注册 XML 映射：

```xml
<mapping resource="com/example/entity/User.hbm.xml"/>
```

## 6. 自动建表策略

### 6.1 hbm2ddl.auto 选项

```xml
<!-- 开发环境：自动更新表结构 -->
<property name="hibernate.hbm2ddl.auto">update</property>

<!-- 测试环境：每次重新创建表 -->
<property name="hibernate.hbm2ddl.auto">create</property>

<!-- 生产环境：验证表结构 -->
<property name="hibernate.hbm2ddl.auto">validate</property>

<!-- 不做任何操作 -->
<property name="hibernate.hbm2ddl.auto">none</property>
```

### 6.2 各选项说明

| 选项 | 说明 | 适用场景 |
|------|------|----------|
| `create` | 删除旧表，创建新表 | 测试环境 |
| `create-drop` | 创建表，SessionFactor关闭时删除 | 单元测试 |
| `update` | 表不存在则创建，存在则更新 | 开发环境 |
| `validate` | 验证表结构，不匹配则报错 | 生产环境 |
| `none` | 不做任何操作 | 生产环境 |

## 7. SQL 显示与格式化

### 7.1 基本配置

```xml
<!-- 显示 SQL -->
<property name="hibernate.show_sql">true</property>

<!-- 格式化 SQL -->
<property name="hibernate.format_sql">true</property>

<!-- 显示注释 -->
<property name="hibernate.use_sql_comments">true</property>
```

### 7.2 输出示例

```sql
Hibernate: 
    /* insert com.example.entity.User */ 
    insert 
    into
        user
        (age, email, username) 
    values
        (?, ?, ?)
```

## 8. 二级缓存配置

### 8.1 启用二级缓存

```xml
<!-- 启用二级缓存 -->
<property name="hibernate.cache.use_second_level_cache">true</property>

<!-- 启用查询缓存 -->
<property name="hibernate.cache.use_query_cache">true</property>

<!-- 指定缓存提供者 -->
<property name="hibernate.cache.region.factory_class">
    org.hibernate.cache.ehcache.EhCacheRegionFactory
</property>
```

### 8.2 Ehcache 配置

**添加依赖**：

```xml
<dependency>
    <groupId>org.hibernate.orm</groupId>
    <artifactId>hibernate-ehcache</artifactId>
    <version>6.2.7.Final</version>
</dependency>

<dependency>
    <groupId>net.sf.ehcache</groupId>
    <artifactId>ehcache</artifactId>
    <version>2.10.9.2</version>
</dependency>
```

**创建 ehcache.xml**：

```xml
<ehcache>
    <defaultCache
        maxElementsInMemory="10000"
        eternal="false"
        timeToIdleSeconds="120"
        timeToLiveSeconds="120"/>
    
    <cache name="com.example.entity.User"
           maxElementsInMemory="5000"
           eternal="false"
           timeToIdleSeconds="300"
           timeToLiveSeconds="600"/>
</ehcache>
```

## 9. 日志配置

### 9.1 使用 Log4j2

**添加依赖**：

```xml
<dependency>
    <groupId>org.apache.logging.log4j</groupId>
    <artifactId>log4j-core</artifactId>
    <version>2.20.0</version>
</dependency>

<dependency>
    <groupId>org.apache.logging.log4j</groupId>
    <artifactId>log4j-slf4j-impl</artifactId>
    <version>2.20.0</version>
</dependency>
```

**创建 log4j2.xml**：

```xml
<?xml version="1.0" encoding="UTF-8"?>
<Configuration status="WARN">
    <Appenders>
        <Console name="Console" target="SYSTEM_OUT">
            <PatternLayout pattern="%d{HH:mm:ss.SSS} [%t] %-5level %logger{36} - %msg%n"/>
        </Console>
    </Appenders>
    
    <Loggers>
        <!-- Hibernate SQL 日志 -->
        <Logger name="org.hibernate.SQL" level="DEBUG" additivity="false">
            <AppenderRef ref="Console"/>
        </Logger>
        
        <!-- Hibernate 参数绑定日志 -->
        <Logger name="org.hibernate.type.descriptor.sql.BasicBinder" level="TRACE" additivity="false">
            <AppenderRef ref="Console"/>
        </Logger>
        
        <Root level="INFO">
            <AppenderRef ref="Console"/>
        </Root>
    </Loggers>
</Configuration>
```

## 10. 完整配置示例

```xml
<?xml version="1.0" encoding="UTF-8"?>
<!DOCTYPE hibernate-configuration PUBLIC
        "-//Hibernate/Hibernate Configuration DTD 3.0//EN"
        "http://www.hibernate.org/dtd/hibernate-configuration-3.0.dtd">

<hibernate-configuration>
    <session-factory>
        <!-- ==================== -->
        <!-- 数据库连接配置         -->
        <!-- ==================== -->
        <property name="hibernate.connection.driver_class">com.mysql.cj.jdbc.Driver</property>
        <property name="hibernate.connection.url">
            jdbc:mysql://localhost:3306/hibernate_demo?useSSL=false&amp;serverTimezone=Asia/Shanghai
        </property>
        <property name="hibernate.connection.username">root</property>
        <property name="hibernate.connection.password">123456</property>

        <!-- ==================== -->
        <!-- 数据库方言            -->
        <!-- ==================== -->
        <property name="hibernate.dialect">org.hibernate.dialect.MySQLDialect</property>

        <!-- ==================== -->
        <!-- 连接池配置（HikariCP）-->
        <!-- ==================== -->
        <property name="hibernate.connection.provider_class">
            com.zaxxer.hikari.hibernate.HikariConnectionProvider
        </property>
        <property name="hibernate.hikari.minimumIdle">5</property>
        <property name="hibernate.hikari.maximumPoolSize">20</property>
        <property name="hibernate.hikari.idleTimeout">300000</property>
        <property name="hibernate.hikari.connectionTimeout">20000</property>

        <!-- ==================== -->
        <!-- SQL 显示配置          -->
        <!-- ==================== -->
        <property name="hibernate.show_sql">true</property>
        <property name="hibernate.format_sql">true</property>
        <property name="hibernate.use_sql_comments">true</property>

        <!-- ==================== -->
        <!-- 自动建表策略          -->
        <!-- ==================== -->
        <property name="hibernate.hbm2ddl.auto">update</property>

        <!-- ==================== -->
        <!-- 二级缓存配置          -->
        <!-- ==================== -->
        <property name="hibernate.cache.use_second_level_cache">true</property>
        <property name="hibernate.cache.use_query_cache">true</property>
        <property name="hibernate.cache.region.factory_class">
            org.hibernate.cache.ehcache.EhCacheRegionFactory
        </property>

        <!-- ==================== -->
        <!-- 其他配置              -->
        <!-- ==================== -->
        <!-- 批量操作大小 -->
        <property name="hibernate.jdbc.batch_size">50</property>
        
        <!-- 获取连接超时时间 -->
        <property name="hibernate.timeout">10</property>

        <!-- ==================== -->
        <!-- 实体类映射            -->
        <!-- ==================== -->
        <mapping class="com.example.entity.User"/>
        <mapping class="com.example.entity.Product"/>
        <mapping class="com.example.entity.Order"/>
    </session-factory>
</hibernate-configuration>
```

## 12. 总结

### 12.1 核心知识点

✅ **数据库连接**：驱动、URL、用户名、密码  
✅ **数据库方言**：根据不同数据库选择  
✅ **连接池**：C3P0、HikariCP  
✅ **自动建表**：create、update、validate、none  
✅ **二级缓存**：Ehcache 集成  
✅ **日志配置**：Log4j2、SQL 显示  

### 12.2 最佳实践

📌 生产环境使用 HikariCP 连接池  
📌 开发环境开启 SQL 显示，生产环境关闭  
📌 使用 validate 或 none 作为自动建表策略  
📌 合理配置连接池参数  
📌 使用外部配置文件管理敏感信息
