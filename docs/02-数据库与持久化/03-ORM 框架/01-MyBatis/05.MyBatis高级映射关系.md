# MyBatis 高级映射关系

**学习目标：**
> * 掌握一对一映射（association）的使用
> * 掌握一对多映射（collection）的使用
> * 理解多对多映射的实现方式
> * 掌握延迟加载的配置与使用
> * 理解 MyBatis 缓存机制（一级缓存、二级缓存）

## 1. 一对一映射（Association）

### 1.1 场景说明

一个用户对应一个身份证，一个订单对应一个用户。

### 1.2 数据库设计

```sql
-- 用户表
CREATE TABLE `user` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `username` VARCHAR(50),
  `email` VARCHAR(100)
);

-- 身份证表
CREATE TABLE `id_card` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `card_no` VARCHAR(20),
  `user_id` INT UNIQUE,
  FOREIGN KEY (`user_id`) REFERENCES `user`(`id`)
);
```

### 1.3 实体类

```java
@Data
public class User {
    private Integer id;
    private String username;
    private String email;
    private IdCard idCard;  // 一对一
}

@Data
public class IdCard {
    private Integer id;
    private String cardNo;
    private Integer userId;
}
```

### 1.4 映射方式

#### 方式一：嵌套结果（推荐）

通过一次 JOIN 查询获取所有数据。

```xml
<resultMap id="UserWithIdCardResultMap" type="User">
    <id property="id" column="user_id"/>
    <result property="username" column="username"/>
    <result property="email" column="email"/>
    
    <!-- 一对一映射 -->
    <association property="idCard" javaType="IdCard">
        <id property="id" column="card_id"/>
        <result property="cardNo" column="card_no"/>
        <result property="userId" column="user_id"/>
    </association>
</resultMap>

<select id="findUserWithIdCard" resultMap="UserWithIdCardResultMap">
    SELECT u.id AS user_id, u.username, u.email,
           c.id AS card_id, c.card_no, c.user_id
    FROM user u
    LEFT JOIN id_card c ON u.id = c.user_id
    WHERE u.id = #{id}
</select>
```

**优点**：
* 只执行一次 SQL，性能好
* 适合数据量小的场景

**缺点**：
* SQL 较复杂
* 如果关联表数据多，会产生大量冗余数据

#### 方式二：嵌套查询

分两次查询，先查主表，再查关联表。

```xml
<resultMap id="UserWithIdCardResultMap" type="User">
    <id property="id" column="id"/>
    <result property="username" column="username"/>
    <result property="email" column="email"/>
    
    <!-- 一对一映射：嵌套查询 -->
    <association property="idCard" 
                 column="id" 
                 select="findIdCardByUserId"/>
</resultMap>

<!-- 查询用户 -->
<select id="findUserWithIdCard" resultMap="UserWithIdCardResultMap">
    SELECT * FROM user WHERE id = #{id}
</select>

<!-- 查询身份证 -->
<select id="findIdCardByUserId" resultType="IdCard">
    SELECT * FROM id_card WHERE user_id = #{userId}
</select>
```

**优点**：
* SQL 简单清晰
* 可以按需加载（延迟加载）

**缺点**：
* 执行多次 SQL（N+1 问题）
* 性能相对较差

## 2. 一对多映射（Collection）

### 2.1 场景说明

一个用户有多个订单，一个部门有多个员工。

### 2.2 数据库设计

```sql
-- 用户表
CREATE TABLE `user` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `username` VARCHAR(50),
  `email` VARCHAR(100)
);

-- 订单表
CREATE TABLE `orders` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `order_no` VARCHAR(50),
  `amount` DECIMAL(10,2),
  `user_id` INT,
  FOREIGN KEY (`user_id`) REFERENCES `user`(`id`)
);
```

### 2.3 实体类

```java
@Data
public class User {
    private Integer id;
    private String username;
    private String email;
    private List<Order> orders;  // 一对多
}

@Data
public class Order {
    private Integer id;
    private String orderNo;
    private BigDecimal amount;
    private Integer userId;
}
```

### 2.4 映射方式

#### 方式一：嵌套结果

```xml
<resultMap id="UserWithOrdersResultMap" type="User">
    <id property="id" column="user_id"/>
    <result property="username" column="username"/>
    <result property="email" column="email"/>
    
    <!-- 一对多映射 -->
    <collection property="orders" ofType="Order">
        <id property="id" column="order_id"/>
        <result property="orderNo" column="order_no"/>
        <result property="amount" column="amount"/>
        <result property="userId" column="user_id"/>
    </collection>
</resultMap>

<select id="findUserWithOrders" resultMap="UserWithOrdersResultMap">
    SELECT u.id AS user_id, u.username, u.email,
           o.id AS order_id, o.order_no, o.amount, o.user_id
    FROM user u
    LEFT JOIN orders o ON u.id = o.user_id
    WHERE u.id = #{id}
</select>
```

**注意**：
* `<collection>` 用于一对多映射
* `ofType` 指定集合中元素的类型
* 必须正确区分主表和从表的列名（使用别名）

#### 方式二：嵌套查询

```xml
<resultMap id="UserWithOrdersResultMap" type="User">
    <id property="id" column="id"/>
    <result property="username" column="username"/>
    <result property="email" column="email"/>
    
    <!-- 一对多映射：嵌套查询 -->
    <collection property="orders" 
                column="id" 
                select="findOrdersByUserId"/>
</resultMap>

<!-- 查询用户 -->
<select id="findUserWithOrders" resultMap="UserWithOrdersResultMap">
    SELECT * FROM user WHERE id = #{id}
</select>

<!-- 查询订单 -->
<select id="findOrdersByUserId" resultType="Order">
    SELECT * FROM orders WHERE user_id = #{userId}
</select>
```

## 3. 多对多映射

### 3.1 场景说明

一个学生可以选多门课程，一门课程可以被多个学生选择。

### 3.2 数据库设计

```sql
-- 学生表
CREATE TABLE `student` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(50)
);

-- 课程表
CREATE TABLE `course` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `course_name` VARCHAR(100)
);

-- 中间表
CREATE TABLE `student_course` (
  `student_id` INT,
  `course_id` INT,
  PRIMARY KEY (`student_id`, `course_id`),
  FOREIGN KEY (`student_id`) REFERENCES `student`(`id`),
  FOREIGN KEY (`course_id`) REFERENCES `course`(`id`)
);
```

### 3.3 实体类

```java
@Data
public class Student {
    private Integer id;
    private String name;
    private List<Course> courses;  // 多对多
}

@Data
public class Course {
    private Integer id;
    private String courseName;
    private List<Student> students;  // 多对多
}
```

### 3.4 映射实现

```xml
<resultMap id="StudentWithCoursesResultMap" type="Student">
    <id property="id" column="student_id"/>
    <result property="name" column="student_name"/>
    
    <!-- 多对多映射 -->
    <collection property="courses" ofType="Course">
        <id property="id" column="course_id"/>
        <result property="courseName" column="course_name"/>
    </collection>
</resultMap>

<select id="findStudentWithCourses" resultMap="StudentWithCoursesResultMap">
    SELECT s.id AS student_id, s.name AS student_name,
           c.id AS course_id, c.course_name
    FROM student s
    LEFT JOIN student_course sc ON s.id = sc.student_id
    LEFT JOIN course c ON sc.course_id = c.id
    WHERE s.id = #{id}
</select>
```

**说明**：
* 多对多本质上是通过中间表实现的两个一对多
* 查询时需要 JOIN 三张表

## 4. 延迟加载（懒加载）

### 4.1 概念

延迟加载是指在需要时才加载关联数据，而不是一次性加载所有数据。

### 4.2 配置方式

#### 全局配置

在 `mybatis-config.xml` 中配置：

```xml
<settings>
    <!-- 开启延迟加载 -->
    <setting name="lazyLoadingEnabled" value="true"/>
    <!-- 关闭积极加载 -->
    <setting name="aggressiveLazyLoading" value="false"/>
</settings>
```

#### 局部配置

在 association 或 collection 中配置：

```xml
<!-- 立即加载 -->
<association property="idCard" 
             column="id" 
             select="findIdCardByUserId"
             fetchType="eager"/>

<!-- 延迟加载 -->
<association property="idCard" 
             column="id" 
             select="findIdCardByUserId"
             fetchType="lazy"/>
```

**fetchType 属性**：
* `lazy`：延迟加载
* `eager`：立即加载

### 4.3 使用示例

```xml
<resultMap id="UserWithOrdersResultMap" type="User">
    <id property="id" column="id"/>
    <result property="username" column="username"/>
    
    <!-- 延迟加载订单 -->
    <collection property="orders" 
                column="id" 
                select="findOrdersByUserId"
                fetchType="lazy"/>
</resultMap>
```

**测试**：

```java
@Test
public void testLazyLoad() {
    SqlSession session = sqlSessionFactory.openSession();
    UserMapper mapper = session.getMapper(UserMapper.class);
    
    // 只查询用户，不查询订单
    User user = mapper.findUserWithOrders(1);
    System.out.println("用户名：" + user.getUsername());
    
    // 只有调用 getOrders() 时才查询订单
    List<Order> orders = user.getOrders();  // 此时才执行 SQL
    System.out.println("订单数：" + orders.size());
    
    session.close();
}
```

**日志输出**：
```
DEBUG - SELECT * FROM user WHERE id = ?  -- 第一次查询
用户名：张三
DEBUG - SELECT * FROM orders WHERE user_id = ?  -- 第二次查询（延迟加载）
订单数：3
```

### 4.4 注意事项

⚠️ 延迟加载必须在 SqlSession 关闭前访问关联数据  
⚠️ 序列化对象时可能触发延迟加载  
⚠️ 谨慎使用，避免 N+1 查询问题

## 5. MyBatis 缓存机制

### 5.1 缓存架构

```
┌─────────────────────────────────────┐
│         二级缓存（Namespace）         │
│  （多个 SqlSession 共享，可选）       │
└─────────────────────────────────────┘
              ↓
┌─────────────────────────────────────┐
│      一级缓存（SqlSession）          │
│  （同一个 SqlSession 内有效）        │
└─────────────────────────────────────┘
              ↓
┌─────────────────────────────────────┐
│           数据库                      │
└─────────────────────────────────────┘
```

### 5.2 一级缓存

#### 特点

* **默认开启**，无法关闭
* **会话级别**：同一个 SqlSession 内有效
* **存储内容**：查询结果
* **失效条件**：
  - SqlSession 关闭
  - SqlSession 清空缓存（clearCache）
  - 执行增删改操作

#### 测试

```java
@Test
public void testFirstLevelCache() {
    SqlSession session = sqlSessionFactory.openSession();
    UserMapper mapper = session.getMapper(UserMapper.class);
    
    // 第一次查询：从数据库获取
    User user1 = mapper.findById(1);
    System.out.println("第一次查询：" + user1);
    
    // 第二次查询：从缓存获取（不执行 SQL）
    User user2 = mapper.findById(1);
    System.out.println("第二次查询：" + user2);
    
    // 验证是同一个对象
    System.out.println("是否同一对象：" + (user1 == user2));
    
    session.close();
}
```

**日志输出**：
```
DEBUG - SELECT * FROM user WHERE id = ?  -- 第一次查询
第一次查询：User(id=1, username=张三, ...)
第二次查询：User(id=1, username=张三, ...)  -- 无 SQL，从缓存获取
是否同一对象：true
```

#### 清空缓存

```java
// 方式一：手动清空
session.clearCache();

// 方式二：执行增删改操作
mapper.insert(newUser);  // 自动清空缓存
```

### 5.3 二级缓存

#### 特点

* **需要手动开启**
* **Namespace 级别**：同一个 Mapper 的所有 SqlSession 共享
* **存储内容**：数据副本（序列化）
* **实现方式**：需要实体类实现 Serializable 接口

#### 开启步骤

**步骤 1**：全局配置

```xml
<settings>
    <!-- 开启二级缓存 -->
    <setting name="cacheEnabled" value="true"/>
</settings>
```

**步骤 2**：在 Mapper XML 中启用

```xml
<mapper namespace="com.example.mapper.UserMapper">
    
    <!-- 启用二级缓存 -->
    <cache/>
    
    <!-- SQL 语句 -->
    <select id="findAll" resultType="User">
        SELECT * FROM user
    </select>
    
</mapper>
```

**步骤 3**：实体类实现 Serializable

```java
@Data
public class User implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private Integer id;
    private String username;
    private String email;
}
```

#### 测试

```java
@Test
public void testSecondLevelCache() {
    SqlSession session1 = sqlSessionFactory.openSession();
    UserMapper mapper1 = session1.getMapper(UserMapper.class);
    
    // 第一次查询
    User user1 = mapper1.findById(1);
    System.out.println("第一次查询：" + user1);
    session1.close();  // 关闭会话，数据写入二级缓存
    
    SqlSession session2 = sqlSessionFactory.openSession();
    UserMapper mapper2 = session2.getMapper(UserMapper.class);
    
    // 第二次查询：从二级缓存获取
    User user2 = mapper2.findById(1);
    System.out.println("第二次查询：" + user2);
    session2.close();
}
```

**日志输出**：
```
DEBUG - SELECT * FROM user WHERE id = ?  -- 第一次查询
第一次查询：User(id=1, username=张三, ...)
Cache Hit Ratio [com.example.mapper.UserMapper]: 0.5  -- 缓存命中
第二次查询：User(id=1, username=张三, ...)
```

#### cache 标签配置

```xml
<cache
    eviction="LRU"           <!-- 回收策略 -->
    flushInterval="60000"    <!-- 刷新间隔(ms) -->
    size="512"               <!-- 最大缓存数 -->
    readOnly="true"/>        <!-- 是否只读 -->
```

**属性说明**：

| 属性 | 说明 | 可选值 |
|------|------|--------|
| eviction | 回收策略 | LRU、FIFO、SOFT、WEAK |
| flushInterval | 刷新间隔 | 毫秒数 |
| size | 缓存数量 | 正整数 |
| readOnly | 是否只读 | true/false |

**回收策略**：
* `LRU`：最近最少使用（默认）
* `FIFO`：先进先出
* `SOFT`：软引用
* `WEAK`：弱引用

#### 禁用二级缓存

**方式一**：在特定语句上禁用

```xml
<select id="findById" resultType="User" useCache="false">
    SELECT * FROM user WHERE id = #{id}
</select>
```

**方式二**：执行增删改时刷新缓存

```xml
<insert id="insert" flushCache="true">
    INSERT INTO user ...
</insert>
```

### 5.4 缓存注意事项

⚠️ **脏读问题**：多表关联查询时，修改其中一张表可能导致缓存不一致  
⚠️ **适用场景**：适合读多写少、对数据实时性要求不高的场景  
⚠️ **生产环境**：建议使用 Redis 等第三方缓存替代 MyBatis 二级缓存

## 6. 完整示例

### 6.1 数据库设计

```sql
-- 用户表
CREATE TABLE `user` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `username` VARCHAR(50),
  `email` VARCHAR(100)
);

-- 订单表
CREATE TABLE `orders` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `order_no` VARCHAR(50),
  `amount` DECIMAL(10,2),
  `user_id` INT,
  FOREIGN KEY (`user_id`) REFERENCES `user`(`id`)
);

-- 订单项表
CREATE TABLE `order_item` (
  `id` INT PRIMARY KEY AUTO_INCREMENT,
  `product_name` VARCHAR(100),
  `quantity` INT,
  `price` DECIMAL(10,2),
  `order_id` INT,
  FOREIGN KEY (`order_id`) REFERENCES `orders`(`id`)
);
```

### 6.2 实体类

```java
@Data
public class User implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private Integer id;
    private String username;
    private String email;
    private List<Order> orders;  // 一对多
}

@Data
public class Order implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private Integer id;
    private String orderNo;
    private BigDecimal amount;
    private Integer userId;
    private User user;  // 一对一
    private List<OrderItem> items;  // 一对多
}

@Data
public class OrderItem implements Serializable {
    private static final long serialVersionUID = 1L;
    
    private Integer id;
    private String productName;
    private Integer quantity;
    private BigDecimal price;
    private Integer orderId;
}
```

### 6.3 Mapper XML

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.example.mapper.UserMapper">
    
    <!-- 启用二级缓存 -->
    <cache eviction="LRU" flushInterval="60000" size="512" readOnly="true"/>
    
    <!-- 用户 ResultMap（包含订单） -->
    <resultMap id="UserWithOrdersResultMap" type="User">
        <id property="id" column="user_id"/>
        <result property="username" column="username"/>
        <result property="email" column="email"/>
        
        <!-- 一对多：订单列表 -->
        <collection property="orders" ofType="Order">
            <id property="id" column="order_id"/>
            <result property="orderNo" column="order_no"/>
            <result property="amount" column="amount"/>
            <result property="userId" column="user_id"/>
            
            <!-- 一对多：订单项列表 -->
            <collection property="items" ofType="OrderItem">
                <id property="id" column="item_id"/>
                <result property="productName" column="product_name"/>
                <result property="quantity" column="quantity"/>
                <result property="price" column="price"/>
                <result property="orderId" column="order_id"/>
            </collection>
        </collection>
    </resultMap>
    
    <!-- 查询用户及订单、订单项 -->
    <select id="findUserWithOrdersAndItems" resultMap="UserWithOrdersResultMap">
        SELECT u.id AS user_id, u.username, u.email,
               o.id AS order_id, o.order_no, o.amount, o.user_id,
               i.id AS item_id, i.product_name, i.quantity, i.price, i.order_id
        FROM user u
        LEFT JOIN orders o ON u.id = o.user_id
        LEFT JOIN order_item i ON o.id = i.order_id
        WHERE u.id = #{id}
    </select>
    
</mapper>
```

## 8. 总结

### 8.1 核心知识点

✅ **一对一映射**：`<association>` 标签  
✅ **一对多映射**：`<collection>` 标签  
✅ **多对多映射**：通过中间表实现  
✅ **延迟加载**：按需加载关联数据  
✅ **一级缓存**：SqlSession 级别，默认开启  
✅ **二级缓存**：Namespace 级别，需手动开启  

### 8.2 最佳实践

📌 优先使用嵌套结果方式（性能更好）  
📌 合理使用延迟加载避免 N+1 问题  
📌 二级缓存适合读多写少的场景  
📌 生产环境建议使用 Redis 等分布式缓存  
📌 注意缓存一致性问题
