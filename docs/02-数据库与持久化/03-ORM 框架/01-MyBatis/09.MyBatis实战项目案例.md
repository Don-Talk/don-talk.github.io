#### Service 层

```java
@Service
public class UserServiceImpl implements UserService {
    
    @Autowired
    private UserMapper userMapper;
    
    @Autowired
    private UserRoleMapper userRoleMapper;
    
    @Override
    public PageInfo<User> findByPage(int pageNum, int pageSize, 
                                      String keyword, Integer status) {
        PageHelper.startPage(pageNum, pageSize);
        List<User> users = userMapper.findByPage(keyword, status);
        return new PageInfo<>(users);
    }
    
    @Override
    public User findById(Long id) {
        return userMapper.findById(id);
    }
    
    @Transactional
    @Override
    public Long create(User user) {
        // 检查用户名是否存在
        User existUser = userMapper.findByUsername(user.getUsername());
        if (existUser != null) {
            throw new RuntimeException("用户名已存在");
        }
        
        // 加密密码
        user.setPassword(encodePassword(user.getPassword()));
        user.setStatus(1);
        
        userMapper.insert(user);
        return user.getId();
    }
    
    @Transactional
    @Override
    public void update(User user) {
        userMapper.update(user);
    }
    
    @Transactional
    @Override
    public void delete(Long id) {
        userMapper.delete(id);
        // 删除用户角色关联
        userRoleMapper.deleteByUserId(id);
    }
    
    @Transactional
    @Override
    public void batchDelete(List<Long> ids) {
        userMapper.batchDelete(ids);
        userRoleMapper.batchDeleteByUserIds(ids);
    }
    
    private String encodePassword(String password) {
        // 使用 BCrypt 加密
        return new BCryptPasswordEncoder().encode(password);
    }
}
```

#### Controller 层

```java
@RestController
@RequestMapping("/api/users")
public class UserController {
    
    @Autowired
    private UserService userService;
    
    @GetMapping
    public Result<PageInfo<User>> list(
            @RequestParam(defaultValue = "1") int pageNum,
            @RequestParam(defaultValue = "10") int pageSize,
            @RequestParam(required = false) String keyword,
            @RequestParam(required = false) Integer status) {
        
        PageInfo<User> page = userService.findByPage(pageNum, pageSize, keyword, status);
        return Result.success(page);
    }
    
    @GetMapping("/{id}")
    public Result<User> get(@PathVariable Long id) {
        User user = userService.findById(id);
        return Result.success(user);
    }
    
    @PostMapping
    public Result<Long> create(@RequestBody User user) {
        Long id = userService.create(user);
        return Result.success(id);
    }
    
    @PutMapping("/{id}")
    public Result<Void> update(@PathVariable Long id, @RequestBody User user) {
        user.setId(id);
        userService.update(user);
        return Result.success();
    }
    
    @DeleteMapping("/{id}")
    public Result<Void> delete(@PathVariable Long id) {
        userService.delete(id);
        return Result.success();
    }
    
    @DeleteMapping("/batch")
    public Result<Void> batchDelete(@RequestBody List<Long> ids) {
        userService.batchDelete(ids);
        return Result.success();
    }
}
```

### 2.3 核心功能实现

#### 创建订单

```java
@Service
public class OrderService {
    
    @Autowired
    private OrderMapper orderMapper;
    
    @Autowired
    private OrderItemMapper orderItemMapper;
    
    @Autowired
    private ProductMapper productMapper;
    
    @Transactional
    public String createOrder(Long userId, List<OrderItem> items) {
        // 生成订单号
        String orderNo = generateOrderNo();
        
        // 计算总金额
        BigDecimal totalAmount = BigDecimal.ZERO;
        for (OrderItem item : items) {
            // 查询商品信息
            Product product = productMapper.findById(item.getProductId());
            if (product == null) {
                throw new RuntimeException("商品不存在：" + item.getProductId());
            }
            
            // 检查库存
            if (product.getStock() < item.getQuantity()) {
                throw new RuntimeException("库存不足：" + product.getName());
            }
            
            item.setProductName(product.getName());
            item.setPrice(product.getPrice());
            item.setSubtotal(product.getPrice().multiply(new BigDecimal(item.getQuantity())));
            
            totalAmount = totalAmount.add(item.getSubtotal());
            
            // 扣减库存
            productMapper.decreaseStock(item.getProductId(), item.getQuantity());
        }
        
        // 创建订单
        Order order = new Order();
        order.setOrderNo(orderNo);
        order.setUserId(userId);
        order.setTotalAmount(totalAmount);
        order.setStatus(0);  // 待支付
        
        orderMapper.insert(order);
        
        // 创建订单项
        for (OrderItem item : items) {
            item.setOrderId(order.getId());
            orderItemMapper.insert(item);
        }
        
        return orderNo;
    }
    
    private String generateOrderNo() {
        return "ORD" + System.currentTimeMillis() + 
               String.format("%04d", (int)(Math.random() * 10000));
    }
}
```

```xml
<select id="statistics" resultType="map">

        COUNT(*) AS totalCount,
```

```sql
-- 文章表
CREATE TABLE `article` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `title` VARCHAR(200) NOT NULL COMMENT '标题',
  `content` TEXT COMMENT '内容',
  `summary` VARCHAR(500) COMMENT '摘要',
  `cover_image` VARCHAR(200) COMMENT '封面图',
  `author_id` BIGINT NOT NULL COMMENT '作者ID',
  `category_id` BIGINT COMMENT '分类ID',
  `view_count` INT DEFAULT 0 COMMENT '浏览量',
  `like_count` INT DEFAULT 0 COMMENT '点赞数',
  `status` TINYINT DEFAULT 0 COMMENT '状态：0-草稿，1-已发布',
  `create_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `update_time` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  INDEX idx_author (`author_id`),
  INDEX idx_category (`category_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 分类表
CREATE TABLE `category` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(50) NOT NULL COMMENT '分类名称',
  `parent_id` BIGINT DEFAULT 0 COMMENT '父分类ID'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

CREATE TABLE `tag` (
  `id` BIGINT PRIMARY KEY AUTO_INCREMENT,
  `name` VARCHAR(50) NOT NULL UNIQUE COMMENT '标签名称'

-- 文章标签关联表
CREATE TABLE `article_tag` (
  `article_id` BIGINT NOT NULL,
  `tag_id` BIGINT NOT NULL,

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
```

### 3.3 核心功能

#### 文章搜索

```xml
<select id="search" resultType="Article">
    SELECT * FROM article
    <where>
        status = 1
        <if test="keyword != null and keyword != ''">
            AND (
                title LIKE CONCAT('%', #{keyword}, '%')
                OR summary LIKE CONCAT('%', #{keyword}, '%')
            )
        </if>
        <if test="categoryId != null">
            AND category_id = #{categoryId}
        </if>
        <if test="tagId != null">
            AND id IN (
                SELECT article_id FROM article_tag WHERE tag_id = #{tagId}
            )
        </if>
    </where>
    ORDER BY create_time DESC
</select>
```

## 4. 通用工具类

### 4.1 统一返回结果

```java
@Data
public class Result<T> {
    private Integer code;
    private String message;
    private T data;
    
    public static <T> Result<T> success(T data) {
        Result<T> result = new Result<>();
        result.setCode(200);
        result.setMessage("success");
        result.setData(data);
        return result;
    }
    
    public static <T> Result<T> success() {
        return success(null);
    }
    
    public static <T> Result<T> error(String message) {
        Result<T> result = new Result<>();
        result.setCode(500);
        result.setMessage(message);
        return result;
    }
}
```

### 4.2 分页结果

```java
@Data
public class PageResult<T> {
    private Long total;
    private Integer pages;
    private Integer pageNum;
    private Integer pageSize;
    private List<T> list;

        result.setPageNum(pageInfo.getPageNum());
```
