# MyBatis Mapper XML 映射文件

**学习目标：**
> * 掌握 Mapper XML 文件的结构和语法
> * 熟练使用 select、insert、update、delete 标签
> * 理解 parameterType 和 resultType/resultMap 的使用
> * 掌握动态 SQL 的核心标签（if、choose、where、set、foreach）
> * 能够使用 SQL 片段复用代码

---

## 1. Mapper XML 文件结构

### 1.1 基本结构

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.example.mapper.UserMapper">
    
    <!-- SQL 语句配置 -->
    <select id="..." resultType="...">
        ...
    </select>
    
    <insert id="...">
        ...
    </insert>
    
    <update id="...">
        ...
    </update>
    
    <delete id="...">
        ...
    </delete>
    
</mapper>
```

### 1.2 namespace 命名空间

* **作用**：区分不同的 Mapper，必须与 Mapper 接口的全限定名一致
* **示例**：`namespace="com.example.mapper.UserMapper"`

## 2. CRUD 操作详解

### 2.1 select（查询）

#### 2.1.1 基本查询

```xml
<!-- 查询所有用户 -->
<select id="findAll" resultType="User">
    SELECT * FROM user
</select>

<!-- 根据 ID 查询 -->
<select id="findById" parameterType="int" resultType="User">
    SELECT * FROM user WHERE id = #{id}
</select>
```

**属性说明**：
* `id`：方法名，与接口方法对应
* `parameterType`：参数类型（可选）
* `resultType`：返回值类型
* `resultMap`：引用外部 resultMap（复杂映射时使用）

#### 2.1.2 条件查询

```xml
<!-- 根据用户名查询 -->
<select id="findByUsername" parameterType="string" resultType="User">
    SELECT * FROM user WHERE username = #{username}
</select>

<!-- 多条件查询 -->
<select id="findByCondition" resultType="User">
    SELECT * FROM user 
    WHERE age &gt; #{minAge} 
      AND age &lt; #{maxAge}
      AND username LIKE CONCAT('%', #{keyword}, '%')
</select>
```

**注意**：XML 中的特殊字符需要转义
* `>` 写成 `&gt;`
* `<` 写成 `&lt;`
* `&` 写成 `&amp;`

或者使用 CDATA：
```xml
<select id="findByCondition" resultType="User">
    <![CDATA[
        SELECT * FROM user 
        WHERE age > #{minAge} 
          AND age < #{maxAge}
    ]]>
</select>
```

#### 2.1.3 分页查询

```xml
<!-- 分页查询 -->
<select id="findByPage" resultType="User">
    SELECT * FROM user 
    LIMIT #{offset}, #{pageSize}
</select>
```

### 2.2 insert（插入）

#### 2.2.1 基本插入

```xml
<!-- 插入用户 -->
<insert id="insert" parameterType="User">
    INSERT INTO user (username, email, age) 
    VALUES (#{username}, #{email}, #{age})
</insert>
```

#### 2.2.2 获取自增主键

**方式一：useGeneratedKeys（推荐）**

```xml
<insert id="insert" parameterType="User" useGeneratedKeys="true" keyProperty="id">
    INSERT INTO user (username, email, age) 
    VALUES (#{username}, #{email}, #{age})
</insert>
```

**使用**：
```java
User user = new User();
user.setUsername("张三");
user.setEmail("zhangsan@example.com");
user.setAge(25);

mapper.insert(user);
System.out.println("生成的ID：" + user.getId());  // 自动填充
```

**方式二：selectKey**

```xml
<insert id="insert" parameterType="User">
    <selectKey keyProperty="id" resultType="int" order="AFTER">
        SELECT LAST_INSERT_ID()
    </selectKey>
    INSERT INTO user (username, email, age) 
    VALUES (#{username}, #{email}, #{age})
</insert>
```

**order 属性**：
* `AFTER`：在 insert 之后执行（MySQL）
* `BEFORE`：在 insert 之前执行（Oracle）

#### 2.2.3 批量插入

```xml
<!-- 批量插入 -->
<insert id="batchInsert" parameterType="list">
    INSERT INTO user (username, email, age) VALUES
    <foreach collection="list" item="user" separator=",">
        (#{user.username}, #{user.email}, #{user.age})
    </foreach>
</insert>
```

**使用**：
```java
List<User> users = Arrays.asList(
    new User("张三", "zhangsan@example.com", 25),
    new User("李四", "lisi@example.com", 28),
    new User("王五", "wangwu@example.com", 22)
);

mapper.batchInsert(users);
```

### 2.3 update（更新）

#### 2.3.1 基本更新

```xml
<!-- 更新用户 -->
<update id="update" parameterType="User">
    UPDATE user 
    SET username = #{username}, 
        email = #{email}, 
        age = #{age}
    WHERE id = #{id}
</update>
```

#### 2.3.2 选择性更新

只更新非 null 的字段：

```xml
<update id="updateSelective" parameterType="User">
    UPDATE user
    <set>
        <if test="username != null">username = #{username},</if>
        <if test="email != null">email = #{email},</if>
        <if test="age != null">age = #{age},</if>
    </set>
    WHERE id = #{id}
</update>
```

**使用**：
```java
User user = new User();
user.setId(1);
user.setUsername("新名字");  // 只更新用户名

mapper.updateSelective(user);
```

#### 2.3.3 批量更新

```xml
<!-- 批量更新 -->
<update id="batchUpdate">
    <foreach collection="list" item="user" separator=";">
        UPDATE user 
        SET username = #{user.username}, email = #{user.email}
        WHERE id = #{user.id}
    </foreach>
</update>
```

**注意**：需要在 JDBC URL 中添加 `allowMultiQueries=true`

```properties
jdbc.url=jdbc:mysql://localhost:3306/mybatis_demo?allowMultiQueries=true
```

### 2.4 delete（删除）

#### 2.4.1 基本删除

```xml
<!-- 根据 ID 删除 -->
<delete id="delete" parameterType="int">
    DELETE FROM user WHERE id = #{id}
</delete>
```

#### 2.4.2 批量删除

```xml
<!-- 批量删除 -->
<delete id="batchDelete" parameterType="list">
    DELETE FROM user WHERE id IN
    <foreach collection="list" item="id" open="(" separator="," close=")">
        #{id}
    </foreach>
</delete>
```

**使用**：
```java
List<Integer> ids = Arrays.asList(1, 2, 3, 4, 5);
mapper.batchDelete(ids);
```

## 3. 参数传递（parameterType）

### 3.1 单个参数

```xml
<!-- 基本类型 -->
<select id="findById" parameterType="int" resultType="User">
    SELECT * FROM user WHERE id = #{id}
</select>

<!-- 字符串类型 -->
<select id="findByUsername" parameterType="string" resultType="User">
    SELECT * FROM user WHERE username = #{username}
</select>
```

**使用**：
```java
User user = mapper.findById(1);
User user = mapper.findByUsername("张三");
```

### 3.2 多个参数

#### 方式一：使用 @Param 注解（推荐）

**Mapper 接口**：
```java
User findByCondition(@Param("username") String username, 
                     @Param("age") Integer age);
```

**XML**：
```xml
<select id="findByCondition" resultType="User">
    SELECT * FROM user 
    WHERE username = #{username} 
      AND age = #{age}
</select>
```

#### 方式二：使用 Map

**Mapper 接口**：
```java
User findByMap(Map<String, Object> params);
```

**XML**：
```xml
<select id="findByMap" parameterType="map" resultType="User">
    SELECT * FROM user 
    WHERE username = #{username} 
      AND age = #{age}
</select>
```

**使用**：
```java
Map<String, Object> params = new HashMap<>();
params.put("username", "张三");
params.put("age", 25);
mapper.findByMap(params);
```

#### 方式三：使用 JavaBean

**Mapper 接口**：
```java
List<User> findByUser(User user);
```

**XML**：
```xml
<select id="findByUser" parameterType="User" resultType="User">
    SELECT * FROM user 
    WHERE username = #{username} 
      AND age = #{age}
</select>
```

**使用**：
```java
User query = new User();
query.setUsername("张三");
query.setAge(25);
List<User> users = mapper.findByUser(query);
```

### 3.3 #{} 与 ${} 的区别

| 特性 | #{} | ${} |
|------|-----|-----|
| **处理方式** | 预编译处理 | 字符串替换 |
| **安全性** | 防止 SQL 注入 | 存在 SQL 注入风险 |
| **使用场景** | 大多数场景 | 动态表名、列名 |
| **示例** | `WHERE id = #{id}` | `ORDER BY ${columnName}` |

**示例对比**：

```xml
<!-- 安全：使用 #{} -->
<select id="findById" resultType="User">
    SELECT * FROM user WHERE id = #{id}
</select>
<!-- 生成：SELECT * FROM user WHERE id = ? -->

<!-- 危险：使用 ${} -->
<select id="findById" resultType="User">
    SELECT * FROM user WHERE id = ${id}
</select>
<!-- 生成：SELECT * FROM user WHERE id = 1 -->
```

**${} 的安全使用场景**：

```xml
<!-- 动态排序 -->
<select id="findByOrder" resultType="User">
    SELECT * FROM user ORDER BY ${orderBy} ${orderType}
</select>

<!-- 动态表名 -->
<select id="findByTable" resultType="User">
    SELECT * FROM ${tableName} WHERE id = #{id}
</select>
```

## 4. 结果映射（resultType vs resultMap）

### 4.1 resultType（简单映射）

适用于数据库列名与 Java 属性名一致或开启驼峰映射的场景。

```xml
<select id="findAll" resultType="User">
    SELECT * FROM user
</select>
```

**要求**：
* 列名与属性名完全一致，或
* 开启 `mapUnderscoreToCamelCase` 配置

### 4.2 resultMap（复杂映射）

适用于列名与属性名不一致、关联查询等复杂场景。

#### 4.2.1 基本用法

```xml
<!-- 定义 resultMap -->
<resultMap id="UserResultMap" type="User">
    <id property="id" column="user_id"/>
    <result property="username" column="user_name"/>
    <result property="email" column="user_email"/>
    <result property="age" column="user_age"/>
</resultMap>

<!-- 使用 resultMap -->
<select id="findAll" resultMap="UserResultMap">
    SELECT user_id, user_name, user_email, user_age FROM user
</select>
```

**标签说明**：
* `<id>`：主键映射，提高性能
* `<result>`：普通字段映射
* `property`：Java 属性名
* `column`：数据库列名

#### 4.2.2 关联查询映射

**一对一映射**：

```xml
<resultMap id="OrderResultMap" type="Order">
    <id property="id" column="order_id"/>
    <result property="orderNo" column="order_no"/>
    <result property="amount" column="amount"/>
    
    <!-- 关联用户 -->
    <association property="user" javaType="User">
        <id property="id" column="user_id"/>
        <result property="username" column="username"/>
        <result property="email" column="email"/>
    </association>
</resultMap>

<select id="findOrderWithUser" resultMap="OrderResultMap">
    SELECT o.order_id, o.order_no, o.amount,
           u.user_id, u.username, u.email
    FROM orders o
    LEFT JOIN user u ON o.user_id = u.user_id
    WHERE o.order_id = #{id}
</select>
```

**一对多映射**：

```xml
<resultMap id="UserResultMap" type="User">
    <id property="id" column="user_id"/>
    <result property="username" column="username"/>
    
    <!-- 关联订单列表 -->
    <collection property="orders" ofType="Order">
        <id property="id" column="order_id"/>
        <result property="orderNo" column="order_no"/>
        <result property="amount" column="amount"/>
    </collection>
</resultMap>

<select id="findUserWithOrders" resultMap="UserResultMap">
    SELECT u.user_id, u.username,
           o.order_id, o.order_no, o.amount
    FROM user u
    LEFT JOIN orders o ON u.user_id = o.user_id
    WHERE u.user_id = #{id}
</select>
```

## 5. 动态 SQL

动态 SQL 是 MyBatis 的强大特性之一，可以根据条件动态生成 SQL 语句。

### 5.1 if（条件判断）

```xml
<select id="findByCondition" resultType="User">
    SELECT * FROM user WHERE 1=1
    <if test="username != null and username != ''">
        AND username LIKE CONCAT('%', #{username}, '%')
    </if>
    <if test="age != null">
        AND age = #{age}
    </if>
    <if test="email != null and email != ''">
        AND email = #{email}
    </if>
</select>
```

**使用**：
```java
User query = new User();
query.setUsername("张");  // 只设置用户名
// query.setAge(null);    // 不设置年龄

List<User> users = mapper.findByCondition(query);
// 生成 SQL：SELECT * FROM user WHERE 1=1 AND username LIKE '%张%'
```

### 5.2 choose、when、otherwise（多选一）

类似于 Java 的 switch-case-default。

```xml
<select id="findByChoose" resultType="User">
    SELECT * FROM user WHERE 1=1
    <choose>
        <when test="username != null and username != ''">
            AND username = #{username}
        </when>
        <when test="email != null and email != ''">
            AND email = #{email}
        </when>
        <otherwise>
            AND age &gt; 18
        </otherwise>
    </choose>
</select>
```

**逻辑**：
* 如果 username 不为空，按用户名查询
* 否则如果 email 不为空，按邮箱查询
* 否则查询年龄大于 18 的用户

### 5.3 where（智能 WHERE 子句）

自动处理 AND/OR 前缀，避免多余的 AND。

```xml
<select id="findByWhere" resultType="User">
    SELECT * FROM user
    <where>
        <if test="username != null and username != ''">
            AND username LIKE CONCAT('%', #{username}, '%')
        </if>
        <if test="age != null">
            AND age = #{age}
        </if>
    </where>
</select>
```

**优势**：
* 如果没有任何条件，不会生成 WHERE 关键字
* 如果第一个条件是 `AND xxx`，会自动去掉 `AND`

### 5.4 set（智能 SET 子句）

用于更新操作，自动处理逗号。

```xml
<update id="updateSelective" parameterType="User">
    UPDATE user
    <set>
        <if test="username != null">username = #{username},</if>
        <if test="email != null">email = #{email},</if>
        <if test="age != null">age = #{age},</if>
    </set>
    WHERE id = #{id}
</update>
```

**优势**：
* 自动去掉最后一个逗号
* 如果没有任何字段，不会生成 SET 关键字

### 5.5 foreach（循环遍历）

用于批量操作或 IN 查询。

#### 5.5.1 IN 查询

```xml
<select id="findByIds" resultType="User">
    SELECT * FROM user WHERE id IN
    <foreach collection="list" item="id" open="(" separator="," close=")">
        #{id}
    </foreach>
</select>
```

**使用**：
```java
List<Integer> ids = Arrays.asList(1, 2, 3, 4, 5);
List<User> users = mapper.findByIds(ids);
// 生成 SQL：SELECT * FROM user WHERE id IN (1, 2, 3, 4, 5)
```

#### 5.5.2 批量插入

```xml
<insert id="batchInsert">
    INSERT INTO user (username, email, age) VALUES
    <foreach collection="list" item="user" separator=",">
        (#{user.username}, #{user.email}, #{user.age})
    </foreach>
</insert>
```

#### 5.5.3 foreach 属性说明

| 属性 | 说明 | 示例 |
|------|------|------|
| collection | 要遍历的集合 | list、array、map 的 key |
| item | 当前元素的变量名 | user、id |
| index | 当前元素的索引 | 0、1、2 |
| open | 开始符号 | ( |
| close | 结束符号 | ) |
| separator | 分隔符 | , |

### 5.6 trim（自定义字符串处理）

更灵活的字符串处理，可以替代 where 和 set。

```xml
<!-- 替代 where -->
<select id="findByTrim" resultType="User">
    SELECT * FROM user
    <trim prefix="WHERE" prefixOverrides="AND |OR ">
        <if test="username != null">
            AND username = #{username}
        </if>
        <if test="age != null">
            AND age = #{age}
        </if>
    </trim>
</select>

<!-- 替代 set -->
<update id="updateByTrim">
    UPDATE user
    <trim prefix="SET" suffixOverrides=",">
        <if test="username != null">username = #{username},</if>
        <if test="email != null">email = #{email},</if>
    </trim>
    WHERE id = #{id}
</update>
```

**属性说明**：
* `prefix`：添加前缀
* `suffix`：添加后缀
* `prefixOverrides`：去掉的前缀
* `suffixOverrides`：去掉的后缀

## 6. SQL 片段（sql、include）

### 6.1 作用

复用常用的 SQL 片段，减少重复代码。

### 6.2 使用示例

```xml
<!-- 定义 SQL 片段 -->
<sql id="Base_Column_List">
    id, username, email, age, create_time
</sql>

<sql id="Where_Condition">
    <where>
        <if test="username != null and username != ''">
            AND username LIKE CONCAT('%', #{username}, '%')
        </if>
        <if test="age != null">
            AND age = #{age}
        </if>
    </where>
</sql>

<!-- 使用 SQL 片段 -->
<select id="findAll" resultType="User">
    SELECT <include refid="Base_Column_List"/> FROM user
</select>

<select id="findByCondition" resultType="User">
    SELECT <include refid="Base_Column_List"/> 
    FROM user
    <include refid="Where_Condition"/>
</select>
```

### 6.3 注意事项

⚠️ SQL 片段中不要包含 `<where>`、`<set>` 等标签，除非整个片段就是这些标签  
⚠️ 基于单表定义 SQL 片段，不要跨表使用

## 7. 完整示例

### 7.1 Mapper 接口

```java
package com.example.mapper;

import com.example.entity.User;
import org.apache.ibatis.annotations.Param;
import java.util.List;

public interface UserMapper {
    
    // 查询
    List<User> findAll();
    User findById(Integer id);
    List<User> findByCondition(User user);
    
    // 插入
    int insert(User user);
    int batchInsert(List<User> users);
    
    // 更新
    int update(User user);
    int updateSelective(User user);
    
    // 删除
    int delete(Integer id);
    int batchDelete(List<Integer> ids);
}
```

### 7.2 Mapper XML

```xml
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.example.mapper.UserMapper">
    
    <!-- SQL 片段 -->
    <sql id="Base_Column_List">
        id, username, email, age, create_time
    </sql>
    
    <!-- 查询所有 -->
    <select id="findAll" resultType="User">
        SELECT <include refid="Base_Column_List"/> FROM user
    </select>
    
    <!-- 根据 ID 查询 -->
    <select id="findById" resultType="User">
        SELECT <include refid="Base_Column_List"/> 
        FROM user WHERE id = #{id}
    </select>
    
    <!-- 条件查询 -->
    <select id="findByCondition" resultType="User">
        SELECT <include refid="Base_Column_List"/> FROM user
        <where>
            <if test="username != null and username != ''">
                AND username LIKE CONCAT('%', #{username}, '%')
            </if>
            <if test="email != null and email != ''">
                AND email = #{email}
            </if>
            <if test="age != null">
                AND age = #{age}
            </if>
        </where>
    </select>
    
    <!-- 插入 -->
    <insert id="insert" useGeneratedKeys="true" keyProperty="id">
        INSERT INTO user (username, email, age) 
        VALUES (#{username}, #{email}, #{age})
    </insert>
    
    <!-- 批量插入 -->
    <insert id="batchInsert">
        INSERT INTO user (username, email, age) VALUES
        <foreach collection="list" item="user" separator=",">
            (#{user.username}, #{user.email}, #{user.age})
        </foreach>
    </insert>
    
    <!-- 更新 -->
    <update id="update">
        UPDATE user 
        SET username = #{username}, email = #{email}, age = #{age}
        WHERE id = #{id}
    </update>
    
    <!-- 选择性更新 -->
    <update id="updateSelective">
        UPDATE user
        <set>
            <if test="username != null">username = #{username},</if>
            <if test="email != null">email = #{email},</if>
            <if test="age != null">age = #{age},</if>
        </set>
        WHERE id = #{id}
    </update>
    
    <!-- 删除 -->
    <delete id="delete">
        DELETE FROM user WHERE id = #{id}
    </delete>
    
    <!-- 批量删除 -->
    <delete id="batchDelete">
        DELETE FROM user WHERE id IN
        <foreach collection="list" item="id" open="(" separator="," close=")">
            #{id}
        </foreach>
    </delete>
    
</mapper>
```

## 9. 总结

### 9.1 核心知识点

✅ **CRUD 标签**：select、insert、update、delete  
✅ **参数传递**：#{} 预编译、${} 字符串替换  
✅ **结果映射**：resultType 简单映射、resultMap 复杂映射  
✅ **动态 SQL**：if、choose、where、set、foreach、trim  
✅ **SQL 片段**：sql、include 复用代码  

### 9.2 最佳实践

📌 优先使用 #{} 防止 SQL 注入  
📌 使用 where、set 标签简化 SQL 拼接  
📌 使用 SQL 片段减少重复代码  
📌 合理使用 resultMap 处理复杂映射  
📌 批量操作使用 foreach 提高效率
