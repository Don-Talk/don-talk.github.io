```sql
-- ROWNUM 分页（Oracle 12c 之前）
SELECT * FROM (
    SELECT t.*, ROWNUM as rn
    FROM (SELECT * FROM employees ORDER BY hire_date DESC) t
    WHERE ROWNUM <= 20
) WHERE rn > 10;
```

### 04.多表连接深度解析（待创建）

**核心内容**：
1. **连接类型**
   - INNER JOIN

   - RIGHT OUTER JOIN
   - FULL OUTER JOIN
   - CROSS JOIN
   - NATURAL JOIN

2. **连接语法**
   - ANSI SQL 标准
   - Oracle 传统语法（+）

3. **连接优化**
   - 驱动表选择
   - 连接顺序
   - 索引使用

4. **高级连接**
   - 自连接
   - 非等值连接
   - 多列连接

**关键 SQL**：
```sql
-- ANSI 标准连接
SELECT e.name, d.dept_name
FROM employees e
INNER JOIN departments d ON e.dept_id = d.dept_id;

-- 左外连接
SELECT e.name, d.dept_name
FROM employees e
LEFT JOIN departments d ON e.dept_id = d.dept_id;

-- Oracle 传统语法
SELECT e.name, d.dept_name
FROM employees e, departments d
WHERE e.dept_id = d.dept_id(+);

-- 自连接
SELECT e1.name as employee, e2.name as manager
FROM employees e1
LEFT JOIN employees e2 ON e1.manager_id = e2.emp_id;
```

### 05.子查询与相关子查询（待创建）

**核心内容**：
1. **子查询类型**
   - 单行子查询
   - 多行子查询
   - 多列子查询

2. **子查询位置**
   - WHERE 子句
   - FROM 子句（内联视图）
   - SELECT 子句（标量子查询）

3. **相关子查询**
   - EXISTS vs IN
   - 性能对比
   - 优化技巧

4. **高级子查询**
   - 标量子查询缓存
   - 子查询展开
   - 物化子查询

**关键 SQL**：
```sql
-- 单行子查询
SELECT name, salary
FROM employees
WHERE salary > (SELECT AVG(salary) FROM employees);

-- 多行子查询
SELECT name, dept_id
FROM employees
WHERE dept_id IN (
    SELECT dept_id FROM departments WHERE location = 'NEW YORK'
);

-- EXISTS vs IN
SELECT name
FROM employees e
WHERE EXISTS (
    SELECT 1 FROM departments d 
    WHERE d.dept_id = e.dept_id AND d.location = 'NEW YORK'
);

-- 相关子查询
SELECT name, salary,
    (SELECT AVG(salary) FROM employees e2 WHERE e2.dept_id = e1.dept_id) as dept_avg
FROM employees e1;
```

### 06.集合操作（待创建）

**核心内容**：
1. **UNION**：去重合并
2. **UNION ALL**：不去重合并（性能更好）
3. **INTERSECT**：交集
4. **MINUS**：差集
5. **集合操作规则**
   - 列数必须相同
   - 数据类型必须兼容
   - ORDER BY 只能在最后使用

**关键 SQL**：
```sql
-- UNION（去重）
SELECT dept_id FROM employees
UNION
SELECT dept_id FROM departments;

-- UNION ALL（更快）
SELECT dept_id FROM employees
UNION ALL
SELECT dept_id FROM departments;

-- INTERSECT（交集）
SELECT dept_id FROM employees
INTERSECT
SELECT dept_id FROM departments;

-- MINUS（差集）
SELECT dept_id FROM departments
MINUS
SELECT dept_id FROM employees;
```

### 07.分析函数大全（待创建）

**核心内容**：
1. **排名函数**
   - ROW_NUMBER()
   - RANK()
   - DENSE_RANK()
   - NTILE()

2. **偏移函数**
   - LAG()
   - LEAD()

3. **首尾值函数**
   - FIRST_VALUE()
   - LAST_VALUE()

4. **聚合分析函数**
   - SUM() OVER()
   - AVG() OVER()
   - COUNT() OVER()
   - MAX() / MIN() OVER()

5. **窗口子句**
   - ROWS BETWEEN
   - RANGE BETWEEN
   - UNBOUNDED PRECEDING/FOLLOWING

**关键 SQL**：
```sql
-- 排名
SELECT name, salary,
    ROW_NUMBER() OVER (ORDER BY salary DESC) as row_num,
    RANK() OVER (ORDER BY salary DESC) as rank,
    DENSE_RANK() OVER (ORDER BY salary DESC) as dense_rank
FROM employees;

-- 累计求和
SELECT order_date, amount,
    SUM(amount) OVER (ORDER BY order_date) as running_total,
    AVG(amount) OVER (ORDER BY order_date ROWS BETWEEN 6 PRECEDING AND CURRENT ROW) as moving_avg
FROM orders;

-- LAG/LEAD
SELECT name, hire_date,
    LAG(hire_date) OVER (ORDER BY hire_date) as prev_hire,
    LEAD(hire_date) OVER (ORDER BY hire_date) as next_hire
FROM employees;

-- 分区统计
SELECT dept_id, name, salary,
    SUM(salary) OVER (PARTITION BY dept_id) as dept_total,
    ROUND(salary / SUM(salary) OVER (PARTITION BY dept_id) * 100, 2) as pct_of_dept
FROM employees;
```

### 08.层次查询（待创建）

**核心内容**：
1. **CONNECT BY 语法**
2. **START WITH 子句**
3. **PRIOR 关键字**
4. **LEVEL 伪列**
5. **层次函数**
   - SYS_CONNECT_BY_PATH
   - CONNECT_BY_ROOT
   - CONNECT_BY_ISLEAF
   - CONNECT_BY_ISCYCLE
6. **避免循环（NOCYCLE）**
7. **树形结构应用**
   - 组织架构
   - 分类目录
   - 菜单树

**关键 SQL**：
```sql
-- 基本层次查询
SELECT employee_id, manager_id, name, LEVEL
FROM employees
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id;

-- 格式化输出
SELECT LPAD(' ', LEVEL * 2) || name as org_chart, LEVEL
FROM employees
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id;

-- 完整路径
SELECT employee_id, name,
    SYS_CONNECT_BY_PATH(name, ' -> ') as path,
    CONNECT_BY_ROOT name as root_employee,
    CONNECT_BY_ISLEAF as is_leaf
FROM employees
START WITH manager_id IS NULL
CONNECT BY NOCYCLE PRIOR employee_id = manager_id;

-- 查找叶子节点
SELECT name
FROM employees
WHERE CONNECT_BY_ISLEAF = 1
START WITH manager_id IS NULL
CONNECT BY PRIOR employee_id = manager_id;
```
