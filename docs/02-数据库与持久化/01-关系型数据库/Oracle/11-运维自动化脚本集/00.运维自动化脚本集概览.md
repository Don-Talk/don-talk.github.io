## 1. 日常巡检脚本

### 1.1 数据库健康检查

```sql
-- health_check.sql - 数据库健康检查脚本
SET LINESIZE 200
SET PAGESIZE 100
COLUMN status FORMAT A10
COLUMN name FORMAT A20

-- 1. 检查数据库状态
SELECT 
    name,
    open_mode,
    database_role,
    log_mode
FROM v$database;

-- 2. 检查实例状态
SELECT 
    instance_name,
    status,
    startup_time,
    host_name
FROM v$instance;

-- 3. 检查表空间使用率
SELECT 
    tablespace_name,
    ROUND(used_space_mb, 2) as used_mb,
    ROUND(total_space_mb, 2) as total_mb,
    ROUND(used_percent, 2) as used_pct
FROM (
    SELECT 
        d.tablespace_name,
        NVL(SUM(f.bytes)/1024/1024, 0) as free_space_mb,
        SUM(d.bytes)/1024/1024 as total_space_mb,
        (SUM(d.bytes) - NVL(SUM(f.bytes), 0))/1024/1024 as used_space_mb,
        ROUND((SUM(d.bytes) - NVL(SUM(f.bytes), 0)) / SUM(d.bytes) * 100, 2) as used_percent
    FROM dba_data_files d
    LEFT JOIN dba_free_space f ON d.tablespace_name = f.tablespace_name
    GROUP BY d.tablespace_name
)
WHERE used_percent > 80  -- 告警阈值 80%
ORDER BY used_percent DESC;

-- 4. 检查会话数
SELECT 
    COUNT(*) as total_sessions,
    COUNT(CASE WHEN status = 'ACTIVE' THEN 1 END) as active_sessions,
    COUNT(CASE WHEN status = 'INACTIVE' THEN 1 END) as inactive_sessions
FROM v$session;

-- 5. 检查锁等待
SELECT 
    COUNT(*) as lock_wait_count
FROM v$session
WHERE blocking_session IS NOT NULL;

-- 6. 检查长时间运行的事务
SELECT 
    s.sid,
    s.serial#,
    s.username,
    t.start_time,
    ROUND((SYSDATE - TO_DATE(t.start_time, 'MM/DD/YY HH24:MI:SS')) * 24 * 60, 2) as duration_minutes
FROM v$session s, v$transaction t
WHERE s.taddr = t.addr
AND (SYSDATE - TO_DATE(t.start_time, 'MM/DD/YY HH24:MI:SS')) * 24 > 1;  -- 超过1小时

-- 7. 检查失效对象
SELECT 
    owner,
    object_type,
    COUNT(*) as invalid_count
FROM dba_objects
WHERE status = 'INVALID'
GROUP BY owner, object_type
HAVING COUNT(*) > 0;

-- 8. 检查归档日志空间
SELECT 
    destination,
    status,
    error
FROM v$archive_dest
WHERE status != 'INACTIVE';

PROMPT ========================================
PROMPT 健康检查完成！
PROMPT ========================================
```

### 1.2 每日巡检报告

```bash
#!/bin/bash
# daily_check.sh - 每日巡检脚本

export ORACLE_SID=orcl
export ORACLE_HOME=/u01/app/oracle/product/19c/dbhome_1
export PATH=$ORACLE_HOME/bin:$PATH

LOG_FILE="/tmp/oracle_daily_check_$(date +%Y%m%d).log"

echo "========================================" > $LOG_FILE
echo "Oracle 每日巡检报告" >> $LOG_FILE
echo "时间: $(date)" >> $LOG_FILE
echo "数据库: $ORACLE_SID" >> $LOG_FILE
echo "========================================" >> $LOG_FILE

# 执行健康检查
sqlplus -s / as sysdba << EOF >> $LOG_FILE
@health_check.sql
EOF

# 检查告警日志
echo "" >> $LOG_FILE
echo "=== 最近 24 小时告警日志 ===" >> $LOG_FILE
tail -100 $ORACLE_BASE/diag/rdbms/$ORACLE_SID/$ORACLE_SID/trace/alert_$ORACLE_SID.log | grep -i "error\|warning\|ora-" >> $LOG_FILE

# 发送邮件（需要配置 mail）
# mail -s "Oracle Daily Check Report - $ORACLE_SID" dba-team@example.com < $LOG_FILE

echo "巡检报告已生成: $LOG_FILE"
```

## 2. 性能监控脚本

### 2.1 Top SQL 监控

```sql
-- top_sql.sql - 查询最耗资源的 SQL
SET LINESIZE 200
SET PAGESIZE 50

-- Top 10 CPU 消耗 SQL
SELECT * FROM (
    SELECT 
        sql_id,
        executions,
        ROUND(cpu_time/1000000, 2) as cpu_seconds,
        ROUND(elapsed_time/1000000, 2) as elapsed_seconds,
        ROUND(cpu_time/executions/1000000, 3) as avg_cpu_per_exec,
        buffer_gets,
        disk_reads,
        SUBSTR(sql_text, 1, 100) as sql_preview
    FROM v$sql
    WHERE executions > 0
    ORDER BY cpu_time DESC
)
WHERE ROWNUM <= 10;

-- Top 10 逻辑读 SQL
SELECT * FROM (
    SELECT 
        sql_id,
        executions,
        buffer_gets,
        ROUND(buffer_gets/executions) as avg_buffer_gets,
        disk_reads,
        SUBSTR(sql_text, 1, 100) as sql_preview
    FROM v$sql
    WHERE executions > 0
    ORDER BY buffer_gets DESC
)
WHERE ROWNUM <= 10;

-- Top 10 物理读 SQL
SELECT * FROM (
    SELECT 
        sql_id,
        executions,
        disk_reads,
        ROUND(disk_reads/executions) as avg_disk_reads,
        buffer_gets,
        SUBSTR(sql_text, 1, 100) as sql_preview
    FROM v$sql
    WHERE executions > 0
    ORDER BY disk_reads DESC
)
WHERE ROWNUM <= 10;
```

### 2.2 等待事件监控

```sql
-- wait_events.sql - 当前等待事件监控
SET LINESIZE 200

SELECT 
    event,
    wait_class,
    COUNT(*) as session_count,
    AVG(seconds_in_wait) as avg_wait_seconds,
    MAX(seconds_in_wait) as max_wait_seconds
FROM v$session
WHERE wait_class != 'Idle'
GROUP BY event, wait_class
ORDER BY session_count DESC;

-- 历史等待事件（过去 1 小时）
SELECT 
    event,
    wait_class,
    COUNT(*) as samples,
    ROUND(COUNT(*) * 10 / 60, 2) as avg_active_sessions
FROM v$active_session_history
WHERE sample_time > SYSDATE - 1/24
GROUP BY event, wait_class
ORDER BY samples DESC
FETCH FIRST 20 ROWS ONLY;
```

### 2.3 实时性能监控脚本

```bash
#!/bin/bash
# monitor_performance.sh - 实时性能监控

export ORACLE_SID=orcl

while true; do
    clear
    echo "=========================================="
    echo "Oracle 实时性能监控 - $(date)"
    echo "=========================================="
    
    sqlplus -s / as sysdba << EOF
    SET PAGESIZE 50
    SET LINESIZE 150
    
    PROMPT === 活动会话 ===
    SELECT COUNT(*) as active_sessions 
    FROM v\$session 
    WHERE status = 'ACTIVE';
    
    PROMPT === Top 等待事件 ===
    SELECT event, COUNT(*) as cnt 
    FROM v\$session 
    WHERE wait_class != 'Idle' 
    GROUP BY event 
    ORDER BY cnt DESC 
    FETCH FIRST 5 ROWS ONLY;
    
    PROMPT === 系统负载 ===
    SELECT 
        ROUND(value, 2) as load_avg
    FROM v\$sysmetric 
    WHERE metric_name = 'Host CPU Utilization (%)'
    AND group_id = 2;
EOF
    
    sleep 5
done
```

## 3. 备份验证脚本

### 3.1 RMAN 备份检查

```sql
-- check_backup.sql - 检查最近的备份
SET LINESIZE 200

-- 最近的完整备份
SELECT 
    start_time,
    end_time,
    ROUND((end_time - start_time) * 24 * 60, 2) as duration_minutes,
    status,
    input_type,
    ROUND(output_bytes/1024/1024/1024, 2) as backup_size_gb
FROM v$rman_backup_job_details
WHERE start_time > SYSDATE - 7
ORDER BY start_time DESC;

-- 备份文件列表
SELECT 
    handle,
    ROUND(bytes/1024/1024/1024, 2) as size_gb,
    completion_time,
    status
FROM v$backup_piece
WHERE completion_time > SYSDATE - 7
ORDER BY completion_time DESC;

-- 检查是否有失败的备份
SELECT COUNT(*) as failed_backups
FROM v$rman_backup_job_details
WHERE status = 'FAILED'
AND start_time > SYSDATE - 7;
```

### 3.2 备份验证自动化

```bash
#!/bin/bash
# verify_backup.sh - 验证备份完整性

export ORACLE_SID=orcl
export ORACLE_HOME=/u01/app/oracle/product/19c/dbhome_1

LOG_FILE="/tmp/backup_verify_$(date +%Y%m%d).log"

echo "开始验证备份..." > $LOG_FILE

# 验证最新备份
rman target / << EOF >> $LOG_FILE
RUN {
    ALLOCATE CHANNEL ch1 TYPE DISK;
    RESTORE DATABASE VALIDATE;
    RELEASE CHANNEL ch1;
}
EXIT;
EOF

# 检查结果
if grep -q "validation succeeded" $LOG_FILE; then
    echo "✅ 备份验证成功" >> $LOG_FILE
    exit 0
else
    echo "❌ 备份验证失败" >> $LOG_FILE
    # 发送告警邮件
    # mail -s "Backup Verification Failed - $ORACLE_SID" dba-team@example.com < $LOG_FILE
    exit 1
fi
```

## 4. 空间管理脚本

### 4.1 表空间使用分析

```sql
-- tablespace_usage.sql - 表空间使用详细分析
SET LINESIZE 200
COLUMN tablespace_name FORMAT A20

SELECT 
    d.tablespace_name,
    ROUND(SUM(d.bytes)/1024/1024, 2) as total_mb,
    ROUND(NVL(SUM(f.bytes), 0)/1024/1024, 2) as free_mb,
    ROUND((SUM(d.bytes) - NVL(SUM(f.bytes), 0))/1024/1024, 2) as used_mb,
    ROUND((SUM(d.bytes) - NVL(SUM(f.bytes), 0)) / SUM(d.bytes) * 100, 2) as used_pct,
    COUNT(DISTINCT d.file_id) as file_count,
    ROUND(MAX(d.maxbytes)/1024/1024, 2) as max_size_mb,
    CASE 
        WHEN (SUM(d.bytes) - NVL(SUM(f.bytes), 0)) / SUM(d.bytes) * 100 > 90 THEN 'CRITICAL'
        WHEN (SUM(d.bytes) - NVL(SUM(f.bytes), 0)) / SUM(d.bytes) * 100 > 80 THEN 'WARNING'
        ELSE 'OK'
    END as status
FROM dba_data_files d
LEFT JOIN dba_free_space f ON d.tablespace_name = f.tablespace_name
GROUP BY d.tablespace_name
ORDER BY used_pct DESC;

-- 预测表空间耗尽时间
SELECT 
    tablespace_name,
    current_size_mb,
    growth_rate_mb_per_day,
    CASE 
        WHEN growth_rate_mb_per_day > 0 
        THEN ROUND((max_size_mb - current_size_mb) / growth_rate_mb_per_day)
        ELSE NULL
    END as days_until_full
FROM (
    SELECT 
        ts.name as tablespace_name,
        ROUND(SUM(df.bytes)/1024/1024) as current_size_mb,
        ROUND(MAX(df.maxbytes)/1024/1024) as max_size_mb,
        -- 简化：假设每天增长 1%
        ROUND(SUM(df.bytes)/1024/1024 * 0.01) as growth_rate_mb_per_day
    FROM v$tablespace ts, v$datafile df
    WHERE ts.ts# = df.ts#
    GROUP BY ts.name
)
WHERE growth_rate_mb_per_day > 0
ORDER BY days_until_full ASC;
```

### 4.2 大对象查找

```sql
-- find_large_objects.sql - 查找占用空间最大的对象
SET LINESIZE 200

SELECT 
    owner,
    segment_name,
    segment_type,
    ROUND(bytes/1024/1024, 2) as size_mb,
    tablespace_name
FROM dba_segments
WHERE bytes > 100*1024*1024  -- 大于 100MB
ORDER BY bytes DESC
FETCH FIRST 20 ROWS ONLY;

-- 查找增长最快的表
SELECT 
    table_owner,
    table_name,
    SUM(space_used_delta)/1024/1024 as growth_mb_last_7_days
FROM dba_hist_seg_stat
WHERE snap_id IN (
    SELECT snap_id 
    FROM dba_hist_snapshot 
    WHERE begin_interval_time > SYSDATE - 7
)
GROUP BY table_owner, table_name
HAVING SUM(space_used_delta) > 0
ORDER BY growth_mb_last_7_days DESC
FETCH FIRST 20 ROWS ONLY;
```

## 5. 故障诊断脚本

### 5.1 死锁检测

```sql
-- detect_deadlock.sql - 检测死锁
SET LINESIZE 200

SELECT 
    l1.sid as waiting_sid,
    s1.username as waiting_user,
    s1.program as waiting_program,
    l1.type as lock_type,
    l1.lmode as requested_mode,
    l2.sid as blocking_sid,
    s2.username as blocking_user,
    s2.program as blocking_program,
    l2.lmode as held_mode
FROM v$lock l1, v$lock l2, v$session s1, v$session s2
WHERE l1.block = 0
AND l2.block = 1
AND l1.id1 = l2.id1
AND l1.id2 = l2.id2
AND l1.sid = s1.sid
AND l2.sid = s2.sid
AND l1.request > 0;

-- 查看死锁详细信息
SELECT 
    sid,
    serial#,
    username,
    status,
    blocking_session,
    final_blocking_session,
    event,
    seconds_in_wait
FROM v$session
WHERE blocking_session IS NOT NULL
OR sid IN (SELECT blocking_session FROM v$session);
```

### 5.2 性能问题快速诊断

```sql
-- quick_diag.sql - 快速性能诊断
SET LINESIZE 200

PROMPT === 1. 系统负载 ===
SELECT 
    ROUND(value, 2) as cpu_utilization_pct
FROM v$sysmetric 
WHERE metric_name = 'Host CPU Utilization (%)'
AND group_id = 2;

PROMPT === 2. 活动会话 ===
SELECT 
    COUNT(*) as total_sessions,
    COUNT(CASE WHEN status = 'ACTIVE' THEN 1 END) as active_sessions
FROM v$session;

PROMPT === 3. Top 等待事件 ===
SELECT 
    event,
    COUNT(*) as session_count
FROM v$session
WHERE wait_class != 'Idle'
GROUP BY event
ORDER BY session_count DESC
FETCH FIRST 5 ROWS ONLY;

PROMPT === 4. 最长运行 SQL ===
SELECT 
    sql_id,
    ROUND(elapsed_time/1000000, 2) as elapsed_seconds,
    SUBSTR(sql_text, 1, 80) as sql_text
FROM v$sql
ORDER BY elapsed_time DESC
FETCH FIRST 5 ROWS ONLY;

PROMPT === 5. 临时表空间使用 ===
SELECT 
    tablespace_name,
    ROUND(used_blocks * 8192 / 1024 / 1024, 2) as used_mb
FROM v$sort_segment;

PROMPT === 诊断完成 ===
```

### 5.3 自动化故障收集

```bash
#!/bin/bash
# collect_diagnostics.sh - 自动收集故障诊断信息

export ORACLE_SID=orcl
export TIMESTAMP=$(date +%Y%m%d_%H%M%S)
DIAG_DIR="/tmp/oracle_diag_$TIMESTAMP"

mkdir -p $DIAG_DIR

echo "收集诊断信息到: $DIAG_DIR"

# 1. 导出 AWR 报告
sqlplus -s / as sysdba << EOF
@?/rdbms/admin/awrrpt.sql
EOF

# 2. 导出 ASH 报告
sqlplus -s / as sysdba << EOF
@?/rdbms/admin/ashrpt.sql
EOF

# 3. 收集系统统计
sqlplus -s / as sysdba << EOF > $DIAG_DIR/system_stats.txt
@quick_diag.sql
EOF

# 4. 收集告警日志
cp $ORACLE_BASE/diag/rdbms/$ORACLE_SID/$ORACLE_SID/trace/alert_$ORACLE_SID.log \
   $DIAG_DIR/alert_log_recent.log

# 5. 压缩打包
tar czf oracle_diag_$TIMESTAMP.tar.gz $DIAG_DIR

echo "诊断包已生成: oracle_diag_$TIMESTAMP.tar.gz"
echo "请发送给 DBA 团队进行分析"
```
