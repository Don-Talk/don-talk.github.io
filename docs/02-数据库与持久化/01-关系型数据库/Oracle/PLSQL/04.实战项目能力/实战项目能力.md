## 批量数据导入、更新、删除

在企业级应用中，经常需要处理大量数据的批量操作。以下是几种常见的批量数据处理场景。

### 批量数据导入

#### 1. 从CSV文件导入数据

```sql
-- 创建外部表用于读取CSV文件
CREATE DIRECTORY data_dir AS '/path/to/data/files';

CREATE TABLE external_employees (
    employee_id NUMBER,
    first_name VARCHAR2(50),
    last_name VARCHAR2(50),
    email VARCHAR2(100),
    hire_date DATE,
    job_id VARCHAR2(10),
    salary NUMBER
)
ORGANIZATION EXTERNAL (
    TYPE ORACLE_LOADER
    DEFAULT DIRECTORY data_dir
    ACCESS PARAMETERS (
        RECORDS DELIMITED BY NEWLINE
        FIELDS TERMINATED BY ','
        OPTIONALLY ENCLOSED BY '"'
        MISSING FIELD VALUES ARE NULL
    )
    LOCATION ('employees.csv')
);

-- 批量插入到目标表
CREATE OR REPLACE PROCEDURE import_employees AS
    v_batch_size CONSTANT PLS_INTEGER := 1000;
    v_count PLS_INTEGER := 0;
    
    TYPE emp_table_type IS TABLE OF external_employees%ROWTYPE INDEX BY PLS_INTEGER;
    v_employees emp_table_type;
    
    CURSOR ext_cursor IS
        SELECT * FROM external_employees;
BEGIN
    OPEN ext_cursor;
    
    LOOP
        FETCH ext_cursor BULK COLLECT INTO v_employees LIMIT v_batch_size;
        
        IF v_employees.COUNT > 0 THEN
            FORALL i IN 1..v_employees.COUNT
                INSERT INTO employees (
                    employee_id, first_name, last_name, email, 
                    hire_date, job_id, salary
                ) VALUES (
                    v_employees(i).employee_id,
                    v_employees(i).first_name,
                    v_employees(i).last_name,
                    v_employees(i).email,
                    v_employees(i).hire_date,
                    v_employees(i).job_id,
                    v_employees(i).salary
                );
            
            v_count := v_count + v_employees.COUNT;
            COMMIT;
        END IF;
        
        EXIT WHEN v_employees.COUNT = 0;
    END LOOP;
    
    CLOSE ext_cursor;
    
    DBMS_OUTPUT.PUT_LINE('Successfully imported ' || v_count || ' employees');
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20001, 'Import failed: ' || SQLERRM);
END;
/
```

### 批量数据更新

#### 1. 基于条件的批量更新

```sql
CREATE OR REPLACE PROCEDURE batch_update_salaries(
    p_department_id NUMBER,
    p_increase_percentage NUMBER
) AS
    TYPE emp_id_table_type IS TABLE OF employees.employee_id%TYPE INDEX BY PLS_INTEGER;
    TYPE salary_table_type IS TABLE OF employees.salary%TYPE INDEX BY PLS_INTEGER;
    
    v_emp_ids emp_id_table_type;
    v_new_salaries salary_table_type;
    
    v_start_time TIMESTAMP;
    v_end_time TIMESTAMP;
BEGIN
    v_start_time := SYSTIMESTAMP;
    
    -- 计算新的薪资
    SELECT employee_id, salary * (1 + p_increase_percentage/100)
    BULK COLLECT INTO v_emp_ids, v_new_salaries
    FROM employees
    WHERE department_id = p_department_id;
    
    -- 批量更新
    IF v_emp_ids.COUNT > 0 THEN
        FORALL i IN 1..v_emp_ids.COUNT
            UPDATE employees 
            SET salary = v_new_salaries(i),
                last_updated = SYSDATE
            WHERE employee_id = v_emp_ids(i);
        
        COMMIT;
        
        v_end_time := SYSTIMESTAMP;
        DBMS_OUTPUT.PUT_LINE('Updated ' || v_emp_ids.COUNT || ' employees in ' || 
                           EXTRACT(SECOND FROM (v_end_time - v_start_time)) || ' seconds');
    ELSE
        DBMS_OUTPUT.PUT_LINE('No employees found for department: ' || p_department_id);
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20002, 'Batch update failed: ' || SQLERRM);
END;
/
```

```sql
CREATE OR REPLACE PROCEDURE sync_employee_departments AS

    );
    
    TYPE dept_mapping_table_type IS TABLE OF dept_mapping_type INDEX BY PLS_INTEGER;
```

### 批量数据删除

#### 1. 安全的大批量删除

```sql
CREATE OR REPLACE PROCEDURE cleanup_old_logs(
    p_retention_days NUMBER DEFAULT 90
) AS
    v_batch_size CONSTANT PLS_INTEGER := 5000;
    v_deleted_count PLS_INTEGER := 0;
    v_total_deleted PLS_INTEGER := 0;
    v_cutoff_date DATE;
BEGIN
    v_cutoff_date := SYSDATE - p_retention_days;
    
    LOOP
        -- 分批删除旧日志
        DELETE FROM application_logs
        WHERE log_date < v_cutoff_date
        AND ROWNUM <= v_batch_size;
        
        v_deleted_count := SQL%ROWCOUNT;
        v_total_deleted := v_total_deleted + v_deleted_count;
        
        -- 提交每批删除
        IF v_deleted_count > 0 THEN
            COMMIT;
            DBMS_OUTPUT.PUT_LINE('Deleted ' || v_deleted_count || ' logs, total: ' || v_total_deleted);
        END IF;
        
        -- 如果没有更多记录要删除，退出循环
        EXIT WHEN v_deleted_count = 0;
        
        -- 短暂暂停以避免过度占用资源
        DBMS_LOCK.SLEEP(1);
    END LOOP;
    
    DBMS_OUTPUT.PUT_LINE('Cleanup completed. Total deleted: ' || v_total_deleted);
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20004, 'Log cleanup failed: ' || SQLERRM);
END;
/
```

#### 2. 带备份的删除操作

```sql
CREATE OR REPLACE PROCEDURE archive_and_delete_records(
    p_table_name VARCHAR2,
    p_condition VARCHAR2,
    p_archive_table VARCHAR2
) AS
    v_sql VARCHAR2(4000);
    v_count NUMBER;
BEGIN
    -- 首先将符合条件的数据复制到归档表
    v_sql := 'INSERT INTO ' || p_archive_table || ' SELECT * FROM ' || p_table_name || ' WHERE ' || p_condition;
    EXECUTE IMMEDIATE v_sql;
    
    v_count := SQL%ROWCOUNT;
    
    IF v_count > 0 THEN
        -- 然后从原表中删除
        v_sql := 'DELETE FROM ' || p_table_name || ' WHERE ' || p_condition;
        EXECUTE IMMEDIATE v_sql;
        
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Archived and deleted ' || v_count || ' records from ' || p_table_name);
    ELSE
        DBMS_OUTPUT.PUT_LINE('No records matched the condition');
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20005, 'Archive and delete failed: ' || SQLERRM);
END;
/
```

## 数据清洗、数据迁移

数据清洗和迁移是ETL（Extract, Transform, Load）过程中的核心环节。

### 数据清洗

#### 1. 标准化数据格式

```sql
CREATE OR REPLACE PROCEDURE clean_employee_data AS
    v_updated_count PLS_INTEGER := 0;
BEGIN
    -- 清理电话号码格式
    UPDATE employees
    SET phone_number = REGEXP_REPLACE(phone_number, '[^0-9+]', '')
    WHERE phone_number IS NOT NULL
    AND REGEXP_LIKE(phone_number, '[^0-9+]');
    
    v_updated_count := v_updated_count + SQL%ROWCOUNT;
    
    -- 标准化电子邮件地址为小写
    UPDATE employees
    SET email = LOWER(email)
    WHERE email != LOWER(email);
    
    v_updated_count := v_updated_count + SQL%ROWCOUNT;
    
    -- 清理姓名中的多余空格
    UPDATE employees
    SET first_name = TRIM(REGEXP_REPLACE(first_name, '\s+', ' ')),
        last_name = TRIM(REGEXP_REPLACE(last_name, '\s+', ' '))
    WHERE first_name LIKE '% %' OR last_name LIKE '% %';
    
    v_updated_count := v_updated_count + SQL%ROWCOUNT;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Cleaned ' || v_updated_count || ' records');
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20006, 'Data cleaning failed: ' || SQLERRM);
END;
/
```

```sql
CREATE OR REPLACE PROCEDURE handle_missing_values AS
    v_avg_salary NUMBER;
    v_median_hire_date DATE;
BEGIN
    -- 计算平均薪资用于填充缺失值
    SELECT AVG(salary) INTO v_avg_salary
    FROM employees
    WHERE salary IS NOT NULL;
    
    -- 用平均值填充缺失的薪资
    UPDATE employees
    SET salary = v_avg_salary
    WHERE salary IS NULL;
    
    -- 查找并标记异常的薪资值（超过平均值3倍标准差）
    UPDATE employees
    SET salary_flag = 'OUTLIER'
    WHERE salary > (v_avg_salary + 3 * STDDEV(salary))
    OR salary < (v_avg_salary - 3 * STDDEV(salary));
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Handled missing values and flagged outliers');
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20007, 'Missing value handling failed: ' || SQLERRM);
END;
/
```

#### 3. 数据去重

```sql
CREATE OR REPLACE PROCEDURE remove_duplicate_records AS
    TYPE rowid_table_type IS TABLE OF ROWID INDEX BY PLS_INTEGER;
    v_duplicate_rowids rowid_table_type;
    
    CURSOR dup_cursor IS
        SELECT ROWID
        FROM (
            SELECT ROWID,
                   ROW_NUMBER() OVER (
                       PARTITION BY first_name, last_name, email 
                       ORDER BY employee_id
                   ) as rn
            FROM employees
        )
        WHERE rn > 1;
BEGIN
    OPEN dup_cursor;
    FETCH dup_cursor BULK COLLECT INTO v_duplicate_rowids;
    CLOSE dup_cursor;
    
    IF v_duplicate_rowids.COUNT > 0 THEN
        FORALL i IN 1..v_duplicate_rowids.COUNT
            DELETE FROM employees WHERE ROWID = v_duplicate_rowids(i);
        
        COMMIT;
        DBMS_OUTPUT.PUT_LINE('Removed ' || v_duplicate_rowids.COUNT || ' duplicate records');
    ELSE
        DBMS_OUTPUT.PUT_LINE('No duplicates found');
    END IF;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20008, 'Duplicate removal failed: ' || SQLERRM);
END;
/
```

### 数据迁移

#### 1. 跨表数据迁移

```sql
CREATE OR REPLACE PROCEDURE migrate_legacy_data AS
    v_batch_size CONSTANT PLS_INTEGER := 1000;
    v_migrated_count PLS_INTEGER := 0;
    
    TYPE legacy_emp_type IS RECORD (
        id NUMBER,
        name VARCHAR2(100),
        dept VARCHAR2(50),
        sal NUMBER,
        hire_dt DATE
    );
    
    TYPE legacy_emp_table_type IS TABLE OF legacy_emp_type INDEX BY PLS_INTEGER;
    v_legacy_emps legacy_emp_table_type;
    
    CURSOR legacy_cursor IS
        SELECT emp_id, emp_name, department, salary, hire_date
        FROM legacy_employees
        WHERE migration_status = 'PENDING';
BEGIN
    OPEN legacy_cursor;
    
    LOOP
        FETCH legacy_cursor BULK COLLECT INTO v_legacy_emps LIMIT v_batch_size;
        
        IF v_legacy_emps.COUNT > 0 THEN
            -- 转换并插入到新表
            FORALL i IN 1..v_legacy_emps.COUNT
                INSERT INTO employees (
                    employee_id,
                    first_name,
                    last_name,
                    department_id,
                    salary,
                    hire_date,
                    migration_status
                ) VALUES (
                    v_legacy_emps(i).id,
                    SUBSTR(v_legacy_emps(i).name, 1, INSTR(v_legacy_emps(i).name, ' ') - 1),
                    SUBSTR(v_legacy_emps(i).name, INSTR(v_legacy_emps(i).name, ' ') + 1),
                    get_department_id(v_legacy_emps(i).dept), -- 假设有一个函数转换部门名到ID
                    v_legacy_emps(i).sal,
                    v_legacy_emps(i).hire_dt,
                    'MIGRATED'
                );
            
            -- 更新源表状态
            FORALL i IN 1..v_legacy_emps.COUNT
                UPDATE legacy_employees
                SET migration_status = 'MIGRATED',
                    migration_date = SYSDATE
                WHERE emp_id = v_legacy_emps(i).id;
            
            v_migrated_count := v_migrated_count + v_legacy_emps.COUNT;
            COMMIT;
        END IF;
        
        EXIT WHEN v_legacy_emps.COUNT = 0;
    END LOOP;
    
    CLOSE legacy_cursor;
    
    DBMS_OUTPUT.PUT_LINE('Migration completed. Total migrated: ' || v_migrated_count);
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20009, 'Data migration failed: ' || SQLERRM);
END;
/
```

#### 2. 带验证的数据迁移

```sql
CREATE OR REPLACE PROCEDURE validated_data_migration AS
    v_success_count PLS_INTEGER := 0;
    v_error_count PLS_INTEGER := 0;
    
    TYPE validation_result_type IS RECORD (
        source_id NUMBER,
        is_valid BOOLEAN,
        error_message VARCHAR2(4000)
    );
    
    TYPE validation_table_type IS TABLE OF validation_result_type INDEX BY PLS_INTEGER;
    v_validations validation_table_type;
BEGIN
    -- 验证源数据
    FOR rec IN (SELECT * FROM source_table WHERE migration_status = 'PENDING') LOOP
        DECLARE
            v_is_valid BOOLEAN := TRUE;
            v_error_msg VARCHAR2(4000) := NULL;
        BEGIN
            -- 业务规则验证
            IF rec.amount < 0 THEN
                v_is_valid := FALSE;
                v_error_msg := 'Amount cannot be negative';
            ELSIF rec.transaction_date > SYSDATE THEN
                v_is_valid := FALSE;
                v_error_msg := 'Transaction date cannot be in future';
            END IF;
            
            -- 如果验证通过，则迁移数据
            IF v_is_valid THEN
                INSERT INTO target_table (
                    transaction_id,
                    amount,
                    transaction_date,
                    description
                ) VALUES (
                    rec.id,
                    rec.amount,
                    rec.transaction_date,
                    rec.description
                );
                
                UPDATE source_table
                SET migration_status = 'SUCCESS'
                WHERE id = rec.id;
                
                v_success_count := v_success_count + 1;
            ELSE
                UPDATE source_table
                SET migration_status = 'FAILED',
                    error_message = v_error_msg
                WHERE id = rec.id;
                
                v_error_count := v_error_count + 1;
            END IF;
        END;
    END LOOP;
    
    COMMIT;
    DBMS_OUTPUT.PUT_LINE('Migration completed. Success: ' || v_success_count || ', Errors: ' || v_error_count);
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE_APPLICATION_ERROR(-20010, 'Validated migration failed: ' || SQLERRM);
END;
/
```

## 报表统计存储过程

报表统计是企业应用中的重要功能，通常需要复杂的聚合计算。

### 销售报表统计

```sql
CREATE OR REPLACE PACKAGE sales_report_pkg AS
    TYPE sales_record_type IS RECORD (
        product_name VARCHAR2(100),
        category VARCHAR2(50),
        total_quantity NUMBER,
        total_amount NUMBER,
        avg_unit_price NUMBER,
        sale_count NUMBER
    );
    
    TYPE sales_table_type IS TABLE OF sales_record_type INDEX BY PLS_INTEGER;
    
    PROCEDURE generate_daily_sales_report(
        p_report_date DATE,
        p_result OUT sales_table_type
    );
    
    PROCEDURE generate_monthly_summary(
        p_year NUMBER,
        p_month NUMBER
    );
    
    FUNCTION get_top_products(
        p_limit NUMBER DEFAULT 10
    ) RETURN sales_table_type;
END sales_report_pkg;
/

CREATE OR REPLACE PACKAGE BODY sales_report_pkg AS
    PROCEDURE generate_daily_sales_report(
        p_report_date DATE,
        p_result OUT sales_table_type
    ) AS
    BEGIN
        SELECT 
            p.product_name,
            c.category_name,
            SUM(oi.quantity) as total_quantity,
            SUM(oi.quantity * oi.unit_price) as total_amount,
            AVG(oi.unit_price) as avg_unit_price,
            COUNT(*) as sale_count
        BULK COLLECT INTO p_result
        FROM order_items oi
        JOIN products p ON oi.product_id = p.product_id
        JOIN categories c ON p.category_id = c.category_id
        JOIN orders o ON oi.order_id = o.order_id
        WHERE TRUNC(o.order_date) = TRUNC(p_report_date)
        GROUP BY p.product_name, c.category_name
        ORDER BY total_amount DESC;
    END;
    
    PROCEDURE generate_monthly_summary(
        p_year NUMBER,
        p_month NUMBER
    ) AS
        v_start_date DATE;
        v_end_date DATE;
        v_total_sales NUMBER;
        v_total_orders NUMBER;
        v_avg_order_value NUMBER;
    BEGIN
        v_start_date := TO_DATE(p_year || '-' || p_month || '-01', 'YYYY-MM-DD');
        v_end_date := LAST_DAY(v_start_date);
        
        SELECT 
            SUM(oi.quantity * oi.unit_price),
            COUNT(DISTINCT o.order_id),
            AVG(order_total)
        INTO 
            v_total_sales,
            v_total_orders,
            v_avg_order_value
        FROM orders o
        JOIN order_items oi ON o.order_id = oi.order_id
        WHERE o.order_date BETWEEN v_start_date AND v_end_date;
        
        -- 保存月度汇总结果
        INSERT INTO monthly_sales_summary (
            year, month, total_sales, total_orders, avg_order_value, created_date
        ) VALUES (
            p_year, p_month, v_total_sales, v_total_orders, v_avg_order_value, SYSDATE
        );
        
        COMMIT;
        
        DBMS_OUTPUT.PUT_LINE('Monthly summary generated for ' || p_year || '-' || p_month);
        DBMS_OUTPUT.PUT_LINE('Total Sales: $' || ROUND(v_total_sales, 2));
        DBMS_OUTPUT.PUT_LINE('Total Orders: ' || v_total_orders);
        DBMS_OUTPUT.PUT_LINE('Average Order Value: $' || ROUND(v_avg_order_value, 2));
    END;
    
    FUNCTION get_top_products(
        p_limit NUMBER DEFAULT 10
    ) RETURN sales_table_type AS
        v_result sales_table_type;
    BEGIN
        SELECT 
            p.product_name,
            c.category_name,
            SUM(oi.quantity) as total_quantity,
            SUM(oi.quantity * oi.unit_price) as total_amount,
            AVG(oi.unit_price) as avg_unit_price,
            COUNT(*) as sale_count
        BULK COLLECT INTO v_result
        FROM order_items oi
        JOIN products p ON oi.product_id = p.product_id
        JOIN categories c ON p.category_id = c.category_id
        GROUP BY p.product_name, c.category_name
        ORDER BY total_amount DESC
        FETCH FIRST p_limit ROWS ONLY;
        
        RETURN v_result;
    END;
END sales_report_pkg;
/
```

### 财务报表统计

```sql
CREATE OR REPLACE PROCEDURE generate_financial_report(
    p_start_date DATE,
    p_end_date DATE,
    p_report_type VARCHAR2 DEFAULT 'SUMMARY' -- SUMMARY, DETAILED, COMPARATIVE
) AS
    TYPE financial_record_type IS RECORD (
        account_code VARCHAR2(20),
        account_name VARCHAR2(100),
        debit_amount NUMBER,
        credit_amount NUMBER,
        balance NUMBER,
        transaction_count NUMBER
    );
    
    TYPE financial_table_type IS TABLE OF financial_record_type INDEX BY PLS_INTEGER;
    v_financial_data financial_table_type;
BEGIN
    IF p_report_type = 'SUMMARY' THEN
        -- 生成摘要报表
        SELECT 
            a.account_code,
            a.account_name,
            NVL(SUM(CASE WHEN t.transaction_type = 'DEBIT' THEN t.amount END), 0) as debit_amount,
            NVL(SUM(CASE WHEN t.transaction_type = 'CREDIT' THEN t.amount END), 0) as credit_amount,
            NVL(SUM(CASE WHEN t.transaction_type = 'DEBIT' THEN t.amount END), 0) - 
            NVL(SUM(CASE WHEN t.transaction_type = 'CREDIT' THEN t.amount END), 0) as balance,
            COUNT(*) as transaction_count
        BULK COLLECT INTO v_financial_data
        FROM accounts a
        LEFT JOIN transactions t ON a.account_id = t.account_id
            AND t.transaction_date BETWEEN p_start_date AND p_end_date
        GROUP BY a.account_code, a.account_name
        ORDER BY a.account_code;
        
    ELSIF p_report_type = 'DETAILED' THEN
        -- 生成详细报表
        SELECT 
            a.account_code,
            a.account_name,
            t.transaction_date,
            t.description,
            t.debit_amount,
            t.credit_amount,
            t.running_balance
        BULK COLLECT INTO v_financial_data
        FROM accounts a
        JOIN transactions t ON a.account_id = t.account_id
        WHERE t.transaction_date BETWEEN p_start_date AND p_end_date
        ORDER BY a.account_code, t.transaction_date;
    END IF;
    
    -- 将报表结果保存到报表历史表
    FOR i IN 1..v_financial_data.COUNT LOOP
        INSERT INTO report_history (
            report_type, start_date, end_date, account_code, 
            account_name, debit_amount, credit_amount, balance, 
            transaction_count, generated_date
        ) VALUES (
            p_report_type, p_start_date, p_end_date,
            v_financial_data(i).account_code,
            v_financial_data(i).account_name,
            v_financial_data(i).debit_amount,
            v_financial_data(i).credit_amount,
            v_financial_data(i).balance,
            v_financial_data(i).transaction_count,
            SYSDATE
        );
    END LOOP;
    
    COMMIT;
    
    DBMS_OUTPUT.PUT_LINE('Financial report generated: ' || p_report_type);
    DBMS_OUTPUT.PUT_LINE('Period: ' || TO_CHAR(p_start_date, 'YYYY-MM-DD') || ' to ' || TO_CHAR(p_end_date, 'YYYY-MM-DD'));
    DBMS_OUTPUT.PUT_LINE('Records processed: ' || v_financial_data.COUNT);
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
```

```sql
    v_sql := v_sql || ' ORDER BY changed_at DESC';

CREATE OR REPLACE PROCEDURE audit_statistics(
    p_days_back NUMBER DEFAULT 30
) AS
    TYPE stat_record_type IS RECORD (
        table_name VARCHAR2(30),
        operation_type VARCHAR2(10),
        operation_count NUMBER,
        unique_users NUMBER
    );
    
    TYPE stat_table_type IS TABLE OF stat_record_type INDEX BY PLS_INTEGER;
    v_stats stat_table_type;
BEGIN
    SELECT 
        table_name,
        operation_type,
        COUNT(*) as operation_count,
        COUNT(DISTINCT changed_by) as unique_users
    BULK COLLECT INTO v_stats
    FROM audit_log
    WHERE changed_at >= SYSDATE - p_days_back
    GROUP BY table_name, operation_type
    ORDER BY operation_count DESC;
    
    DBMS_OUTPUT.PUT_LINE('Audit Statistics for last ' || p_days_back || ' days:');
    DBMS_OUTPUT.PUT_LINE('Table Name | Operation | Count | Unique Users');
    DBMS_OUTPUT.PUT_LINE('-----------|-----------|-------|-------------');
    
    FOR i IN 1..v_stats.COUNT LOOP
        DBMS_OUTPUT.PUT_LINE(
            RPAD(v_stats(i).table_name, 12) || ' | ' ||
            RPAD(v_stats(i).operation_type, 9) || ' | ' ||
            LPAD(v_stats(i).operation_count, 5) || ' | ' ||
            LPAD(v_stats(i).unique_users, 12)
        );
    END LOOP;
END;
```
