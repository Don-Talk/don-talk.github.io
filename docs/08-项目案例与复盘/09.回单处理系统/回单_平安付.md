# 平安付回单处理模块

## 一、功能概述

平安付（PAP）回单处理模块实现了从平安付 SFTP 服务器自动下载回单文件，上传到 OBS 对象存储，并将流水号与回单文件的对应关系保存到数据库的完整流程。

### 核心功能
- ✅ 定时任务每 5 分钟自动扫描 SFTP 服务器
- ✅ 根据文件名提取流水号
- ✅ 下载 PDF 回单文件
- ✅ 上传到华为云 OBS
- ✅ 记录流水号与回单文件的对应关系
- ✅ 异步并发处理，提高吞吐量
- ✅ 完善的异常处理和状态追踪

### 2.3 并发设计

**线程池配置**：`receiptTaskExecutor`
- 核心线程数：5
- 最大线程数：10
- 队列容量：100
- 拒绝策略：CallerRunsPolicy（由调用线程执行）

**为什么需要线程池？**
1. 控制并发度，避免同时创建过多 SFTP 连接
2. 提高性能，线程复用减少创建开销
3. 隔离任务，不影响其他业务
4. 保证任务不丢失

## 三、数据库设计

### 3.1 表结构

```sql
-- 创建数据库
CREATE DATABASE IF NOT EXISTS dt_finance DEFAULT CHARACTER SET utf8mb4 COLLATE utf8mb4_general_ci;

USE dt_finance;

-- 流水号回单文件关系表
CREATE TABLE IF NOT EXISTS `receipt_relation` (
    `id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键ID',
    `serial_number` VARCHAR(100) NOT NULL COMMENT '流水号',
    `channel` VARCHAR(20) NOT NULL COMMENT '回单渠道: PAP-平安付, PAB-平安银行, HSBC-汇丰银行',
    `document_id` VARCHAR(200) NOT NULL COMMENT 'OBS文档ID',
    `file_name` VARCHAR(200) DEFAULT NULL COMMENT '文件名',
    `file_size` BIGINT DEFAULT NULL COMMENT '文件大小(字节)',
    `status` TINYINT NOT NULL DEFAULT 0 COMMENT '状态: 0-待处理, 1-处理中, 2-成功, 3-失败, 4-已经关联回单文件',
    `error_msg` TEXT COMMENT '错误信息',
    `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
    PRIMARY KEY (`id`),
    UNIQUE KEY `uk_serial_channel` (`serial_number`, `channel`),
    KEY `idx_channel` (`channel`),
    KEY `idx_status` (`status`),
    KEY `idx_create_time` (`create_time`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci COMMENT='流水号回单文件关系表';
```

### 3.2 索引说明

- **唯一索引 `uk_serial_channel`**：防止同一流水号重复处理
- **普通索引 `idx_channel`**：按渠道查询优化
- **普通索引 `idx_status`**：按状态查询优化
- **普通索引 `idx_create_time`**：按时间排序优化

```java
package com.dt.finance.config;

import lombok.extern.slf4j.Slf4j;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;
import org.springframework.scheduling.annotation.EnableAsync;
import org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor;

import java.util.concurrent.Executor;
import java.util.concurrent.ThreadPoolExecutor;

/**
 * 线程池配置
 */
@Slf4j
@Configuration
@EnableAsync
public class ThreadPoolConfig {

    /**
     * 回单处理线程池
     * 用于并发处理回单下载、上传等操作
     */
    @Bean("receiptTaskExecutor")
    public Executor receiptTaskExecutor() {
        ThreadPoolTaskExecutor executor = new ThreadPoolTaskExecutor();
        
        // 核心线程数
        executor.setCorePoolSize(5);
        // 最大线程数
        executor.setMaxPoolSize(10);
        // 队列容量
        executor.setQueueCapacity(100);
        // 线程空闲时间（秒）
        executor.setKeepAliveSeconds(60);
        // 线程名称前缀
        executor.setThreadNamePrefix("receipt-task-");
        // 拒绝策略：由调用线程处理
        executor.setRejectedExecutionHandler(new ThreadPoolExecutor.CallerRunsPolicy());
        // 等待所有任务结束后再关闭线程池
        executor.setWaitForTasksToCompleteOnShutdown(true);
        // 等待时间（秒）
        executor.setAwaitTerminationSeconds(60);
        
        executor.initialize();
        log.info("回单处理线程池初始化完成");
        
        return executor;
    }
}
```

```java
package com.dt.finance.common.util;

import com.jcraft.jsch.*;
import lombok.extern.slf4j.Slf4j;

import java.io.ByteArrayOutputStream;
import java.io.InputStream;
import java.util.Properties;
import java.util.Vector;

/**
 * SFTP 工具类
 */
@Slf4j
public class SftpUtil {

    private Session session;
    private ChannelSftp channelSftp;

    /**
     * 连接 SFTP 服务器
     */
    public void connect(String host, int port, String username, String password) throws JSchException {
        JSch jsch = new JSch();
        session = jsch.getSession(username, host, port);
        session.setPassword(password);

        Properties config = new Properties();
        config.put("StrictHostKeyChecking", "no");
        session.setConfig(config);
        session.connect();

        channelSftp = (ChannelSftp) session.openChannel("sftp");
        channelSftp.connect();
        log.info("SFTP 连接成功: {}", host);
    }

    /**
     * 下载文件
     */
    public byte[] downloadFile(String remotePath) throws Exception {
        try (InputStream inputStream = channelSftp.get(remotePath);
             ByteArrayOutputStream outputStream = new ByteArrayOutputStream()) {
            
            byte[] buffer = new byte[1024];
            int bytesRead;
            while ((bytesRead = inputStream.read(buffer)) != -1) {
                outputStream.write(buffer, 0, bytesRead);
            }
            
            log.info("文件下载成功: {}", remotePath);
            return outputStream.toByteArray();
        }
    }

    /**
     * 列出目录下的文件
     */
    public Vector<ChannelSftp.LsEntry> listFiles(String directory) throws SftpException {
        return channelSftp.ls(directory);
    }

    /**
     * 判断文件是否存在
     *
     * @param remotePath 远程文件路径
     * @return 文件是否存在
     */
    public boolean fileExists(String remotePath) {
        try {
            channelSftp.stat(remotePath);
            return true;
        } catch (SftpException e) {
            // 文件不存在或无法访问
            log.debug("文件不存在: {}", remotePath);
            return false;
        }
    }

    /**
     * 关闭连接
     */
    public void disconnect() {
        if (channelSftp != null && channelSftp.isConnected()) {
            channelSftp.disconnect();
        }
        if (session != null && session.isConnected()) {
            session.disconnect();
        }
        log.info("SFTP 连接已关闭");
    }
}
```

```java
package com.dt.finance.common.util;

import com.obs.services.ObsClient;
import com.obs.services.model.PutObjectRequest;
import com.obs.services.model.PutObjectResult;
import lombok.extern.slf4j.Slf4j;

import java.io.ByteArrayInputStream;

/**
 * OBS 工具类（华为云对象存储）
 */
@Slf4j
public class ObsUtil {

    private ObsClient obsClient;
    private String bucketName;

    /**
     * 初始化 OBS 客户端
     */
    public void init(String endPoint, String ak, String sk, String bucketName) {
        this.obsClient = new ObsClient(ak, sk, endPoint);
        this.bucketName = bucketName;
        log.info("OBS 客户端初始化成功");
    }

    /**
     * 上传文件到 OBS
     *
     * @param fileName 文件名
     * @param content  文件内容
     * @return documentId (objectKey)
     */
    public String uploadFile(String fileName, byte[] content) {
        try {
            PutObjectRequest request = new PutObjectRequest(bucketName, fileName, new ByteArrayInputStream(content));
            PutObjectResult result = obsClient.putObject(request);
            
            log.info("文件上传成功: {}, ETag: {}", fileName, result.getEtag());
            return fileName; // 返回 objectKey 作为 documentId
        } catch (Exception e) {
            log.error("文件上传失败: {}", fileName, e);
            throw new RuntimeException("文件上传失败", e);
        }
    }

    /**
     * 关闭客户端
     */
    public void close() {
        if (obsClient != null) {
            try {
                obsClient.close();
                log.info("OBS 客户端已关闭");
            } catch (Exception e) {
                log.error("关闭 OBS 客户端失败", e);
            }
        }
    }
}
```

```java
import org.springframework.scheduling.annotation.Async;
import org.springframework.stereotype.Service;

import java.time.LocalDateTime;
import java.util.UUID;

/**
 * 平安付回单处理服务
 */
@Slf4j
@Service
public class PapReceiptService {
    
    @Autowired
    private ReceiptProperties receiptProperties;
    
    @Autowired
    private ReceiptRelationMapper receiptRelationMapper;
    
    @Autowired
    private ObsUtil obsUtil;
    
    /**
     * 处理单个流水号的回单（异步）
     *
     * @param serialNumber 流水号
     */
    @Async("receiptTaskExecutor")
    public void processReceipt(String serialNumber) {
        log.info("开始处理平安付回单，流水号: {}", serialNumber);
        
        ReceiptRelation relation = new ReceiptRelation();
        relation.setSerialNumber(serialNumber);
        relation.setChannel(ReceiptChannel.PAP.getCode());
        relation.setStatus(ReceiptStatus.PROCESSING.getCode());
        relation.setCreateTime(LocalDateTime.now());
        relation.setUpdateTime(LocalDateTime.now());
        
        SftpUtil sftpUtil = null;
        
        try {
            // 1. 从数据库插入待处理记录
            receiptRelationMapper.insert(relation);
            
            // 2. 连接 SFTP 并下载文件
            sftpUtil = new SftpUtil();
            ReceiptProperties.PapSftp papSftp = receiptProperties.getPapSftp();
            sftpUtil.connect(papSftp.getHost(), papSftp.getPort(), 
                           papSftp.getUsername(), papSftp.getPassword());
            
            String remoteFilePath = papSftp.getRemotePath() + "/" + serialNumber + ".pdf";
            byte[] pdfContent = sftpUtil.downloadFile(remoteFilePath);
            
            // 3. 上传到 OBS
            String fileName = serialNumber + "_" + UUID.randomUUID().toString().substring(0, 8) + ".pdf";
            String documentId = obsUtil.uploadFile(fileName, pdfContent);
            
            // 4. 更新记录
            relation.setDocumentId(documentId);
            relation.setFileName(fileName);
            relation.setFileSize((long) pdfContent.length);
            relation.setStatus(ReceiptStatus.SUCCESS.getCode());
            receiptRelationMapper.updateStatus(relation);
            
            log.info("平安付回单处理成功，流水号: {}, documentId: {}", serialNumber, documentId);
            
        } catch (Exception e) {
            log.error("平安付回单处理失败，流水号: {}", serialNumber, e);
            relation.setStatus(ReceiptStatus.FAILED.getCode());
            relation.setErrorMsg(e.getMessage());
            receiptRelationMapper.updateStatus(relation);
        } finally {
            if (sftpUtil != null) {
                sftpUtil.disconnect();
            }
        }
    }
}
```

### 4.9 定时任务

**文件位置**：`src/main/java/com/dt/finance/task/ReceiptTaskScheduler.java`

```java
package com.dt.finance.task;

import com.dt.finance.common.util.SftpUtil;
import com.dt.finance.config.ReceiptProperties;
import com.dt.finance.service.receipt.PapReceiptService;
import com.jcraft.jsch.ChannelSftp;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.scheduling.annotation.Scheduled;
import org.springframework.stereotype.Component;

import java.util.List;
import java.util.Vector;
import java.util.stream.Collectors;

/**
 * 回单处理定时任务
 */
@Slf4j
@Component
public class ReceiptTaskScheduler {
    
    @Autowired
    private ReceiptProperties receiptProperties;
    
    @Autowired
    private PapReceiptService papReceiptService;
    
    /**
     * 平安付回单下载任务
     * 每5分钟执行一次
     */
    @Scheduled(cron = "0 */5 * * * ?")
    public void schedulePapReceiptDownload() {
        log.info("开始执行平安付回单下载定时任务");
        
        SftpUtil sftpUtil = null;
        
        try {
            // 连接 SFTP
            sftpUtil = new SftpUtil();
            ReceiptProperties.PapSftp papSftp = receiptProperties.getPapSftp();
            sftpUtil.connect(papSftp.getHost(), papSftp.getPort(), 
                           papSftp.getUsername(), papSftp.getPassword());
            
            // 列出远程目录下的所有 PDF 文件
            String remotePath = papSftp.getRemotePath();
            Vector<ChannelSftp.LsEntry> files = sftpUtil.listFiles(remotePath);
            
            // 过滤出 PDF 文件并提取流水号
            List<String> serialNumbers = files.stream()
                .filter(file -> !file.getFilename().equals(".") && !file.getFilename().equals(".."))
                .filter(file -> file.getFilename().endsWith(".pdf"))
                .map(file -> {
                    String filename = file.getFilename();
                    // 去掉 .pdf 后缀，得到流水号
                    return filename.substring(0, filename.lastIndexOf(".pdf"));
                })
                .collect(Collectors.toList());
            
            log.info("发现 {} 个待处理的平安付回单文件", serialNumbers.size());
            
            // 并发处理每个流水号
            for (String serialNumber : serialNumbers) {
                try {
                    papReceiptService.processReceipt(serialNumber);
                } catch (Exception e) {
                    log.error("提交平安付回单处理任务失败，流水号: {}", serialNumber, e);
                }
            }
            
        } catch (Exception e) {
            log.error("平安付回单下载定时任务执行失败", e);
        } finally {
            if (sftpUtil != null) {
                sftpUtil.disconnect();
            }
        }
        
        log.info("平安付回单下载定时任务执行完成");
    }
}
```

```java
package com.dt.finance.controller;

            List<ReceiptRelation> receipts;
            if (channel != null && !channel.isEmpty()) {
                receipts = receiptRelationMapper.findByChannel(channel, limit);
            } else {
                receipts = receiptRelationMapper.findByChannel("PAP", limit);
            }
            
            result.put("code", 200);
            result.put("data", receipts);
        } catch (Exception e) {
            log.error("查询回单列表失败", e);
            result.put("code", 500);
            result.put("message", "查询失败: " + e.getMessage());
        }
        return result;
    }
    
    /**
     * 根据流水号查询回单
     */
    @GetMapping("/{serialNumber}")
    public Map<String, Object> getReceipt(@PathVariable String serialNumber,
                                          @RequestParam String channel) {
        Map<String, Object> result = new HashMap<>();
        try {
            ReceiptRelation receipt = receiptRelationMapper.findBySerialAndChannel(serialNumber, channel);
            if (receipt != null) {
                result.put("code", 200);
                result.put("data", receipt);
            } else {

            log.error("查询回单失败", e);
            result.put("code", 500);
            result.put("message", "查询失败: " + e.getMessage());
        }
        return result;
    }
}
```

## 六、注意事项

### 6.1 并发安全

1. **唯一索引保护**：数据库唯一索引 `(serial_number, channel)` 防止重复处理
2. **线程池隔离**：使用专用线程池，不影响其他业务
3. **资源释放**：SFTP 连接在 finally 块中确保关闭

### 6.2 异常处理

1. 所有异常都会记录到数据库的 `error_msg` 字段
2. 定时任务异常不会影响下次执行
3. 单个流水号处理失败不影响其他流水号

### 6.3 性能优化

1. **异步处理**：使用 `@Async` 异步执行，提高吞吐量
2. **批量提交**：一次性扫描所有文件，批量提交任务
3. **连接复用**：OBS 客户端使用单例，减少连接开销

### 6.4 监控建议

1. 监控线程池的活跃线程数和队列大小
2. 定期检查 `status=3`（失败）的记录
3. 记录处理耗时，优化慢查询
4. 监控 SFTP 连接成功率

## 七、常见问题

### Q1: 如何处理大量文件？

**A**: 当前实现已经支持并发处理。如果文件量特别大（上千个），可以：
1. 增加线程池大小
2. 添加分页扫描逻辑
3. 使用消息队列（RabbitMQ/Kafka）解耦

### Q2: 如何避免重复处理？

**A**: 通过以下方式保证幂等性：
1. 数据库唯一索引约束
2. 处理前先检查是否已存在
3. 文件名添加 UUID 避免 OBS 覆盖

### Q3: SFTP 连接失败怎么办？

**A**: 
1. 检查网络连接和防火墙
2. 验证用户名密码是否正确
3. 查看日志中的详细错误信息
4. 定时任务会在下次继续尝试

### Q4: 如何扩展到其他渠道？

**A**: 
1. 在 `ReceiptChannel` 枚举中添加新渠道
2. 创建对应的 Service 类（如 `PabReceiptService`）
3. 在定时任务中添加新的调度方法
4. 配置相应的连接参数
